93  void assemble(std::istream& input)
94  {
95  if (not fs::exists(output_dir_)) { fs::create_directory(output_dir_); }
96  fs::path work_dir = output_dir_ / ".batches";
97  if (not fs::exists(work_dir)) { fs::create_directory(work_dir); }
98 
99  int batch_number = 0;
100  std::vector<fs::path> batch_dirs;
101  while (input && input.peek() != EOF)
102  {
103  std::clog << "Building batch " << batch_number << std::endl;
104  fs::path batch_dir = work_dir / std::to_string(batch_number);
105  metadata batch_metadata(batch_dir);
106  build_batch(input,
107  batch_metadata,
108  static_cast<document_type>(batch_number * batch_size_));
109  batch_dirs.push_back(std::move(batch_dir));
110  ++batch_number;
111  }
112  std::clog << "Merging " << batch_number << " batches" << std::endl;
113  merger_type merger(output_dir_, batch_dirs, block_size_);
114  merger.merge();
115  auto term_map = build_lexicon(
116  irk::index::terms_path(output_dir_), lexicon_block_size_);
117  term_map.serialize(irk::index::term_map_path(output_dir_));
118  auto title_map = build_lexicon(
119  irk::index::titles_path(output_dir_), lexicon_block_size_);
120  title_map.serialize(irk::index::title_map_path(output_dir_));
121  std::clog << "Success!" << std::endl;
122  }

129  std::istream& build_batch(std::istream& input,
130  metadata batch_metadata,
131  document_type first_id) const
132  {
133  if (not fs::exists(batch_metadata.dir))
134  {
135  fs::create_directory(batch_metadata.dir);
136  }
137 
138  std::ofstream of_doc_ids(batch_metadata.doc_ids.c_str());
139  std::ofstream of_doc_ids_off(batch_metadata.doc_ids_off.c_str());
140  std::ofstream of_doc_counts(batch_metadata.doc_counts.c_str());
141  std::ofstream of_doc_counts_off(batch_metadata.doc_counts_off.c_str());
142  std::ofstream of_terms(batch_metadata.terms.c_str());
143  std::ofstream of_term_doc_freq(batch_metadata.term_doc_freq.c_str());
144  std::ofstream of_titles(batch_metadata.doc_titles.c_str());
145  std::ofstream of_sizes(batch_metadata.doc_sizes.c_str());
146  std::ofstream of_term_occurrences(
147  batch_metadata.term_occurrences.c_str());
148  std::ofstream of_properties(batch_metadata.properties.c_str());
149 
150  builder_type builder(block_size_);
151  std::string line;
152  for (auto processed_documents_ = static_cast<int32_t>(first_id);
153  processed_documents_
154  < batch_size_ + static_cast<int32_t>(first_id);
155  processed_documents_++)
156  {
157  if (not std::getline(input, line)) { break; }
158  auto doc = static_cast<document_type>(processed_documents_);
159  builder.add_document(doc);
160  std::istringstream linestream(line);
161  std::string title;
162  linestream >> title;
163  of_titles << title << '\n';
164  std::string term;
165  while (linestream >> term) { builder.add_term(term); }
166  }
167 
168  builder.sort_terms();
169  builder.write_terms(of_terms);
170  builder.write_document_frequencies(of_term_doc_freq);
171  builder.write_document_ids(of_doc_ids, of_doc_ids_off);
172  builder.write_document_counts(of_doc_counts, of_doc_counts_off);
173  builder.write_document_sizes(of_sizes);
174  builder.write_term_occurrences(of_term_occurrences);
175  builder.write_properties(of_properties);
176  of_titles.close();
177  of_terms.close();
178  auto term_map = build_lexicon(
179  irk::index::terms_path(batch_metadata.dir), lexicon_block_size_);
180  term_map.serialize(irk::index::term_map_path(batch_metadata.dir));
181  auto title_map = build_lexicon(
182  irk::index::titles_path(batch_metadata.dir), lexicon_block_size_);
183  title_map.serialize(irk::index::title_map_path(batch_metadata.dir));
184 
185  return input;
186  }
