111  self_type& moveto(value_type val)
112  {
113  static_assert(delta_encoded, "must be delta encoded to call moveto");

178  void ensure_decoded() const
179  {
180  if (block_ != decoded_block_num_)
181  {
182  auto count = block_ < block_count_ - 1
183  ? block_size_
184  : view_.length_ - ((block_count_ - 1) * block_size_);
185  if constexpr (delta_encoded) { // NOLINT
186  auto preceding = block_ > 0 ? view_.blocks_[block_ - 1].back()
187  : 0_id;
188  view_.codec_.delta_decode(
189  std::begin(view_.blocks_[block_].data()),
190  std::begin(decoded_block_),
191  count,
192  preceding);
193  }
194  else {
195  decoded_block_.resize(block_size_);
196  view_.codec_.decode(std::begin(view_.blocks_[block_].data()),
197  std::begin(decoded_block_),
198  count);
199  }
200  decoded_block_num_ = block_;
201  }

231 class block_list_builder {
232 public:
233  using codec_type = Codec;
234  using value_type = Value;
235  using encoding_device =
236  boost::iostreams::back_insert_device<std::vector<char>>;

242  std::streamsize write(std::ostream& out) const
243  {
244  std::vector<int32_t> absolute_skips;
245  std::vector<value_type> last_values;
246  std::vector<char> encoded_blocks;
247 
248  gsl::index pos = 0;
249  value_type previous_doc(0);
250  const int32_t num_blocks = (values_.size() + block_size_ - 1)
251  / block_size_;
252  for (gsl::index block = 0; block < num_blocks; ++block)
253  {
254  absolute_skips.push_back(pos);
255  const gsl::index begin_idx = block_size_ * block;
256  const gsl::index end_idx = std::min(
257  begin_idx + block_size_, gsl::index(values_.size()));
258 
259  const value_type* begin = values_.data() + begin_idx;
260  const value_type* end = values_.data() + end_idx;
261 
262  encoded_blocks.resize(
263  pos + value_codec_.max_encoded_size(end_idx - begin_idx));
264  if constexpr (delta_encoded) { // NOLINT
265  last_values.push_back(values_[end_idx - 1]);
266  pos += value_codec_.delta_encode(
267  begin, end, &encoded_blocks[pos], previous_doc);
268  previous_doc = last_values.back();
269  }
270  else {
271  pos += value_codec_.encode(begin, end, &encoded_blocks[pos]);
272  }
273  }
274 
275  const std::vector<char> encoded_header = irk::encode(
276  int_codec_, {block_size_, num_blocks});
277  const std::vector<char> encoded_skips = irk::delta_encode(
278  int_codec_, absolute_skips);
279 
280  int list_byte_size = encoded_header.size() + encoded_skips.size() + pos;
281  std::vector<char> encoded_last_vals;
282 
283  if constexpr (delta_encoded) { // NOLINT
284  encoded_last_vals = irk::delta_encode(
285  value_codec_, last_values); // NOLINT
286  list_byte_size += encoded_last_vals.size();
287  } // NOLINT
288  list_byte_size = expanded_size(list_byte_size); // NOLINT
289 
290  auto encoded_list_byte_size = irk::encode(int_codec_, {list_byte_size});
291  out.write(&encoded_list_byte_size[0], encoded_list_byte_size.size());
292  out.write(&encoded_header[0], encoded_header.size());
293  out.write(&encoded_skips[0], encoded_skips.size());
294  if constexpr (delta_encoded) { // NOLINT
295  out.write(
296  &encoded_last_vals[0], encoded_last_vals.size()); // NOLINT
297  }
298  out.write(&encoded_blocks[0], pos); // NOLINT
299 
300  return list_byte_size;
301  }
302 
303 private:
304  int expanded_size(int list_byte_size) const
305  {
306  unsigned int extra_bytes = 1;
307  while (static_cast<unsigned int>(list_byte_size) + extra_bytes
308  >= (1u << (extra_bytes * 7u)))
309  { extra_bytes++; }
310  return static_cast<int>(list_byte_size + extra_bytes);
311  }
312 
313  int32_t block_size_;
314  codec_type value_codec_;
315  std::vector<value_type> values_;
316  irk::vbyte_codec<int32_t> int_codec_;
317 };

358  pos = vb.decode(pos, &skips[0], num_blocks);
359  std::vector<value_type> last_documents(num_blocks);
360  pos = codec_.delta_decode(pos, &last_documents[0], num_blocks);
361 
362  for (int block = 0; block < num_blocks - 1; block++) {
363  std::advance(pos, skips[block]);
364  blocks_.emplace_back(last_documents[block],
365  irk::make_memory_view(pos, skips[block + 1]));
366  }
367  std::advance(pos, skips.back());
368  blocks_.emplace_back(last_documents.back(),
369  irk::make_memory_view(pos, std::distance(pos, std::end(memory_))));
370  }

373  iterator end() const

375  if (length_ == 0) { return begin(); };
376  auto end_pos =
377  (length_ - ((static_cast<int>(blocks_.size()) - 1) * block_size_))
378  % block_size_;
