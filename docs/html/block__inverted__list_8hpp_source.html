107  template<class = std::enable_if_t<delta_encoded>>
108  self_type& moveto(value_type val)
109  {
110  int block = nextgeq_block(block_, val);
111  if (block >= decoded_blocks_.size())
112  {
113  finish();
114  return *this;
115  }
116  pos_ = block == block_ ? pos_ : 0;
117  block_ = block;
118  ensure_decoded();

120  return *this;
121  };

173  void ensure_decoded() const
174  {
175  if (!decoded_blocks_[block_].has_value())
176  {
177  decoded_blocks_[block_] = std::make_optional(
178  std::vector<value_type>(view_.block_size_));
179  if constexpr (delta_encoded)
180  {

217 class block_list_builder {
218 public:
219  using value_type = Value;
220  using encoding_device =
221  boost::iostreams::back_insert_device<std::vector<char>>;
222 
223  block_list_builder(int block_size, irk::any_codec<value_type> value_codec)
224  : block_size_(block_size), value_codec_(value_codec)
225  {}
226 
227  void add(value_type id) { values_.push_back(id); }

229  std::streamsize write(std::ostream& out) const
230  {
231  std::vector<int> absolute_skips;
232  std::vector<value_type> last_values;
233  std::vector<char> encoded_blocks;
234  boost::iostreams::stream<encoding_device> val_stream(
235  boost::iostreams::back_inserter(encoded_blocks));
236 
237  value_type previous_doc(0);
238  const int num_blocks = (values_.size() + block_size_ - 1) / block_size_;
239  for (gsl::index block = 0; block < num_blocks; ++block)
240  {
241  absolute_skips.push_back(encoded_blocks.size());
242  const gsl::index begin_idx = block_size_ * block;
243  const gsl::index end_idx = std::min(
244  begin_idx + block_size_, gsl::index(values_.size()));
245 
246  if constexpr (delta_encoded)
247  { last_values.push_back(values_[end_idx - 1]); }
248 
249  for (gsl::index idx = begin_idx; idx < end_idx; ++idx)
250  {
251  if constexpr (delta_encoded) {
252  const auto delta = values_[idx] - previous_doc;
253  value_codec_.encode(delta, val_stream);
254  previous_doc = values_[idx];
255  } else {
256  value_codec_.encode(values_[idx], val_stream);
257  }
258  }
259  val_stream.flush();
260  }
261 
262  const std::vector<char> encoded_header = irk::encode(
263  {block_size_, num_blocks}, int_codec_);
264  const std::vector<char> encoded_skips = irk::encode_delta(
265  absolute_skips, int_codec_);
266 
267  int list_byte_size = encoded_header.size() + encoded_skips.size()
268  + encoded_blocks.size();
269  std::vector<char> encoded_last_vals;
270 
271  if constexpr (delta_encoded)
272  {
273  encoded_last_vals = irk::encode_delta(last_values, value_codec_);
274  list_byte_size += encoded_last_vals.size();
275  }
276  list_byte_size = expanded_size(list_byte_size);
277 
278  int_codec_.encode(list_byte_size, out);
279  out.write(&encoded_header[0], encoded_header.size());
280  out.write(&encoded_skips[0], encoded_skips.size());
281  if constexpr (delta_encoded)
282  { out.write(&encoded_last_vals[0], encoded_last_vals.size()); }
283  out.write(&encoded_blocks[0], encoded_blocks.size());
284  out.flush();
285 
286  return list_byte_size;
287  }
288 
289 private:
290  int expanded_size(int list_byte_size) const
291  {
292  int extra_bytes = 1;
293  while (list_byte_size + extra_bytes >= (1 << (extra_bytes * 7)))
294  { extra_bytes++; }
295  return list_byte_size + extra_bytes;
296  }
297 
298  int block_size_;
299  any_codec<value_type> value_codec_;
300  std::vector<value_type> values_;
301  irk::varbyte_codec<int> int_codec_;
302 };
