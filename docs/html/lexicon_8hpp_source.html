75  irk::memory_view block_memory_view(int block) const
76  {
77  EXPECTS(block >= 0);
78  EXPECTS(block < block_offsets_.size());
79  auto block_offset = block_offsets_[block];
80  auto next_block_offset = block + 1 < block_offsets_.size()
81  ? block_offsets_[block + 1]
82  : blocks_.size();
83  std::ptrdiff_t size = next_block_offset - block_offset;
84  ENSURES(size > 0);
85  ENSURES(size <= blocks_.size());
86  if constexpr (std::is_same<memory_container,
87  irk::memory_view>::value) // NOLINT
88  {
89  return blocks_.range(block_offset, size);
90  }
91  else { // NOLINT
92  return irk::make_memory_view(&blocks_[block_offset], size);
93  }
94  }

96  std::optional<value_type> index_at(const std::string& key) const
97  {
98  auto block = leading_keys_->seek_le(key);
99  if (not block.has_value()) { return std::nullopt; }
100  auto block_memory = block_memory_view(*block);
101  boost::iostreams::stream<boost::iostreams::basic_array_source<char>>
102  buffer(block_memory.data(), block_memory.size());
103  irk::input_bit_stream bin(buffer);
104 
105  value_type value = leading_indices_[*block];
106  std::string k;
107  codec_.reset();
108  codec_.decode(bin, k);
109  while (k < key) {
110  codec_.decode(bin, k);
111  ++value;
112  }
113  return k == key ? std::make_optional(value) : std::nullopt;
114  }

116  std::string key_at(std::ptrdiff_t index) const
117  {

120  leading_indices_.begin(), leading_indices_.end(), index));
121  auto block = std::distance(leading_indices_.begin(), block_pos);

139  {
140  std::vector<char> header(max_serialized_header_size());
141  vbyte_codec<std::ptrdiff_t> intcodec;
142  auto header_iter = std::begin(header);
143 
144  std::advance(header_iter, intcodec.encode(&count_, header_iter));
145  auto size = block_offsets_.size();
146  std::advance(header_iter, intcodec.encode(&size, header_iter));
147  std::advance(

149

151  { std::advance(header_iter, intcodec.encode(&offset, header_iter)); }
152  for (const auto& index : leading_indices_)
153  { std::advance(header_iter, intcodec.encode(&index, header_iter)); }

335 inline lexicon_view<hutucker_codec<char>>
336 load_lexicon(const irk::memory_view& memory)
337 {
338  // Total header size: everything that will be always read to memory.
339  auto header_size = memory.range(0, sizeof(ptrdiff_t)).as<std::ptrdiff_t>();
340  auto header_memory = memory(sizeof(ptrdiff_t), header_size);
341  irk::vbyte_codec<std::ptrdiff_t> intcodec;
342  auto header_iter = header_memory.begin();
343 
344  // Block metadata
345  std::ptrdiff_t block_count, value_count, keys_per_block;
346  std::vector<std::ptrdiff_t> block_offsets;
347  std::vector<std::ptrdiff_t> leading_indices;
348  header_iter = intcodec.decode(header_iter, &value_count);
349  header_iter = intcodec.decode(header_iter, &block_count);
350  header_iter = intcodec.decode(header_iter, &keys_per_block);
351  for (int idx : boost::irange<int>(0, block_count)) {
352  (void)idx;
353  std::ptrdiff_t offset;
354  header_iter = intcodec.decode(header_iter, &offset);
355  block_offsets.push_back(offset);
356  }
357  for (int idx : boost::irange<int>(0, block_count)) {
358  (void)idx;
359  std::ptrdiff_t first_index;
360  header_iter = intcodec.decode(header_iter, &first_index);
361  leading_indices.push_back(first_index);
362  }
363 
364  // Encoding tree
365  std::size_t tree_size = *reinterpret_cast<const std::size_t*>(
366  &*header_iter);
367  std::advance(header_iter, sizeof(std::size_t));
368  auto tree_end = std::next(header_iter, tree_size);
369  std::vector<char> tree_data(header_iter, tree_end);
370  alphabetical_bst<> encoding_tree(std::move(tree_data));
371  hutucker_codec<char> ht_codec(std::move(encoding_tree));
372 
373  // Block leading values
374  boost::iostreams::stream<boost::iostreams::basic_array_source<char>> buffer(
375  tree_end,
376  header_size - std::distance(header_memory.begin(), tree_end)
377  - sizeof(header_size));
378  irk::input_bit_stream bin(buffer);
379  auto leading_keys = std::make_shared<irk::radix_tree<int>>();
380  auto pcodec = irk::prefix_codec<hutucker_codec<char>>(std::move(ht_codec));
381  for (int idx : boost::irange<int>(0, block_count)) {
382  std::string key;
383  pcodec.decode(bin, key);
384  leading_keys->insert(key, idx);
385  }
386  return lexicon_view<hutucker_codec<char>>(std::move(block_offsets),
387  std::move(leading_indices),
388  memory(header_size, memory.size()),
389  value_count,
390  keys_per_block,
391  leading_keys,
392  std::move(pcodec));
393 }

406 template<class KeyIterator>
407 lexicon<hutucker_codec<char>, std::vector<char>> build_lexicon(
408  KeyIterator keys_begin,
409  KeyIterator keys_end,
410  KeyIterator corpus_begin,
411  KeyIterator corpus_end,
412  int keys_per_block)
413 {
414  EXPECTS(keys_begin != keys_end);
415  EXPECTS(corpus_begin != corpus_end);
416 
417  std::vector<std::size_t> frequencies(256, 0);
418  for (; corpus_begin != corpus_end; ++corpus_begin)
419  {
420  for (const char& ch : *corpus_begin)
421  { ++frequencies[static_cast<unsigned char>(ch)]; }
422  }
423  auto codec = hutucker_codec<char>(frequencies);
424 
425  std::vector<std::ptrdiff_t> block_offsets;
426  std::vector<std::ptrdiff_t> leading_indices;
427  auto leading_keys = std::make_shared<irk::radix_tree<int>>();
428  std::vector<char> blocks;
429  boost::iostreams::stream<
430  boost::iostreams::back_insert_device<std::vector<char>>>
431  buffer(boost::iostreams::back_inserter(blocks));
432  irk::output_bit_stream bout(buffer);
433  auto pcodec = irk::prefix_codec<hutucker_codec<char>>(std::move(codec));
434 
435  std::ptrdiff_t index = 0;
436  int block_idx = 0;
437  while (keys_begin != keys_end) {
438  block_offsets.push_back(blocks.size());
439  leading_indices.push_back(index++);
440  std::string leading_key = *keys_begin++;
441  leading_keys->insert(leading_key, block_idx);
442  pcodec.reset();
443  pcodec.encode(leading_key, bout);
444  for (std::size_t idx_in_block = 1;
445  idx_in_block < keys_per_block && keys_begin != keys_end;
446  ++idx_in_block, ++index, ++keys_begin)
447  {
448  pcodec.encode(*keys_begin, bout);
449  }
450  ++block_idx;
451  bout.flush();
452  }
453  pcodec.reset();
454  return lexicon<irk::hutucker_codec<char>, std::vector<char>>(
455  std::move(block_offsets),
456  std::move(leading_indices),
457  std::move(blocks),
458  index,
459  keys_per_block,
460  leading_keys,
461  std::move(pcodec));
462 }
