147  std::pair<std::size_t, std::size_t> locate(term_id_type term_id,
148  const offset_table<>& offset_table,
149  std::size_t file_size) const
150  {
151  std::size_t offset = offset_table[term_id];
152  std::size_t following = std::size_t(term_id) + 1 < offset_table.size()
153  ? offset_table[term_id + 1]
154  : file_size;
155  return std::make_pair(offset, following - offset);

158  template<class T>
159  std::vector<T> decode_range(term_id_type term_id,
160  const std::vector<char>& data_container,
161  const offset_table<>& offset_table,
162  bool delta) const
163  {
164  if (term_id >= offset_table.size()) {
165  throw std::out_of_range(
166  "Requested term ID out of range; requested: "
167  + std::to_string(term_id) + " but must be [0, "
168  + std::to_string(term_map_.size()) + ")");
169  }
170  auto[offset, range_size] =
171  locate(term_id, offset_table, data_container.size());
172  return delta
173  ? irk::coding::decode_delta<varbyte_codec<T>>(gsl::span<const char>(
174  data_container.data() + offset, range_size))
175  : irk::coding::decode<varbyte_codec<T>>(gsl::span<const char>(
176  data_container.data() + offset, range_size));
177  }

403  template<class ScoreFn = score::tf_idf_scorer>
404  auto posting_range(const std::string& term,
405  ScoreFn score_fn = score::tf_idf_scorer{}) const
406  -> dspr<_posting<document_type, score::score_result_t<ScoreFn, document_type, Freq>>,
407  Freq,
408  ScoreFn>
409  {
410  auto opt = term_id(term);

412  return empty_posting_range<ScoreFn>();
413  }
414  return posting_range(opt.value(), score_fn);
415  }

417  template<class ScoreFn = score::tf_idf_scorer>
418  auto posting_range(
419  term_id_type term_id, ScoreFn score_fn = score::tf_idf_scorer{}) const
420  -> dspr<_posting<document_type,
421  score::score_result_t<ScoreFn, document_type, Freq>>,
422  Freq,
423  ScoreFn>
424  {
425  using Posting = _posting<document_type,
426  score::score_result_t<ScoreFn, document_type, Freq>>;
427  Freq df = term_dfs_[term_id];
428  std::vector<document_type> docs;
429  std::vector<Freq> tfs;
430  if (in_memory_) {
431  docs = decode_range<document_type>(
432  term_id, doc_ids_, doc_ids_off_, true);
433  tfs = decode_range<Freq>(
434  term_id, doc_counts_, doc_counts_off_, false);
435  } else {
436  auto[doc_offset, doc_range_size] =
437  locate(term_id, doc_ids_off_, doc_ids_size_);
438  auto[count_offset, count_range_size] =
439  locate(term_id, doc_counts_off_, doc_counts_size_);
440  docs = irk::coding::decode_delta<varbyte_codec<document_type>>(
441  load_data(
442  index::doc_ids_path(dir_), doc_offset, doc_range_size));
443  tfs = irk::coding::decode<varbyte_codec<Freq>>(load_data(
444  index::doc_counts_path(dir_), count_offset, count_range_size));
445  }
446  return dspr<Posting, Freq, ScoreFn>(
447  std::move(docs), std::move(tfs), df, titles_.size(), score_fn);
448  }
