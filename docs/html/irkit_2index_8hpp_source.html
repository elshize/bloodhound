663  std::pop_heap(heap_.begin(), heap_.end());
664  result.push_back(heap_.back());
665  heap_.pop_back();
666  }
667  return result;
668  }

672  std::vector<fs::path> indices,
673  bool skip_unique = false)
674  : target_dir_(target_dir), skip_unique_(skip_unique)
675  {
676  for (fs::path index_dir : indices) {
677  std::cout << "Loading index " << index_dir << std::endl;
678  indices_.emplace_back(index_dir, false, true);
679  }
680  terms_out_.open(index::terms_path(target_dir));
681  doc_ids_.open(index::doc_ids_path(target_dir));
682  doc_counts_.open(index::doc_counts_path(target_dir));
683  doc_offset_ = 0;
684  count_offset_ = 0;

687  void merge_term(std::vector<Entry>& indices)
688  {
689  if (skip_unique_ && indices.size() == 1) {
690  auto pr = indices.front().index->posting_range(
691  indices.front().current_term_id, score::Count{});
692  if (pr.size() == 1) {
693  std::cerr << "Skipping unique term "
694  << indices.front().current_term() << std::endl;
695  return;
696  }
697  }
698  std::cerr << "Merging term " << term_dfs_.size() << " ("
699  << indices.front().current_term() << ")" << std::endl;
700 
701  // Write the term.
702  terms_out_ << indices.front().current_term() << std::endl;
703 
704  // Sort by shift, i.e., effectively document IDs.
705  std::sort(indices.begin(),
706  indices.end(),
707  [](const Entry& lhs, const Entry& rhs) {
708  return lhs.shift < rhs.shift;
709  });
710 
711  // Merge the posting lists.
712  std::vector<Doc> doc_ids;
713  std::vector<Freq> doc_counts;
714  for (const Entry& e : indices) {
715  auto pr = e.index->posting_range(e.current_term_id, score::Count{});
716  for (auto p : pr) {
717  doc_ids.push_back(p.doc + e.shift);
718  doc_counts.push_back(p.score);
719  }
720  }
721 
722  // Accumulate the term's document frequency.
723  term_dfs_.push_back(doc_ids.size());
724 
725  // Accumulate offsets.
726  doc_ids_off_.push_back(doc_offset_);
727  doc_counts_off_.push_back(count_offset_);
728 
729  // Write documents and counts.
730  std::vector<char> encoded_ids = VarByte<Doc>{}.delta_encode(doc_ids);
731  doc_ids_.write(encoded_ids.data(), encoded_ids.size());
732  std::vector<char> encoded_counts = VarByte<Freq>{}.encode(doc_counts);
733  doc_counts_.write(encoded_counts.data(), encoded_counts.size());
734 
735  // Calc new offsets.
736  doc_offset_ += encoded_ids.size();
737  count_offset_ += encoded_counts.size();
738  }

740  void merge_terms()
741  {
742  // Initialize heap: everyone starts with the first term.
743  Doc shift(0);
744  for (const _Index& index : indices_) {
745  heap_.push_back({0, &index, shift});
746  shift += index.collection_size();
747  }
748 
749  while (!heap_.empty()) {
750  std::vector<Entry> indices_to_merge = indices_with_next_term();
751  merge_term(indices_to_merge);
752  for (const Entry& e : indices_to_merge) {
753  if (std::size_t(e.current_term_id + 1)
754  < e.index->term_count()) {
755  heap_.push_back(
756  Entry{e.current_term_id + 1, e.index, e.shift});
757  std::push_heap(heap_.begin(), heap_.end());
758  }
759  }
760  }
761 
762  // Write offsets
763  VarByte<std::size_t> vb;
764  std::vector<char> encoded_id_offsets = vb.delta_encode(doc_ids_off_);
765  std::vector<char> encoded_count_offsets =
766  vb.delta_encode(doc_counts_off_);
767  std::ofstream idoff(index::doc_ids_off_path(target_dir_));
768  idoff.write(encoded_id_offsets.data(), encoded_id_offsets.size());
769  idoff.close();
770  std::ofstream countoff(index::doc_counts_off_path(target_dir_));
771  countoff.write(
772  encoded_count_offsets.data(), encoded_count_offsets.size());
773  countoff.close();
774 
775  // Write term dfs.
776  std::vector<char> encoded_dfs = VarByte<Freq>{}.delta_encode(term_dfs_);
777  std::ofstream dfs(index::term_doc_freq_path(target_dir_));
778  dfs.write(encoded_dfs.data(), encoded_dfs.size());
779  dfs.close();
780 
781  // Close other streams.
782  terms_out_.close();
783  doc_ids_.close();
784  doc_counts_.close();
785  }
