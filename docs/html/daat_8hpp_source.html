377  const std::vector<Score>& term_weights)
378  {
379 #ifdef STATS
380  std::size_t postings = 0;
381 #endif
382  std::vector<List> essential;
383  std::vector<List> non_essential;
384  for (unsigned int idx = 0; idx < term_postings.size(); ++idx) {
385  if (!term_postings[idx].empty()) {
386  auto max_score =
387  term_postings[idx].max_score * term_weights[idx];
388  essential.push_back({idx, max_score});
389 #ifdef STATS
390  postings += term_postings[idx].length();
391 #endif

393  }
394 #ifdef STATS

396 #endif
397 
398  // Sort by decreasing max score so that we can pop_back
399  // to move the smallest essential to the non-essential list.
400  std::sort(essential.begin(),
401  essential.end(),
402  [](const List& a, const List& b) {
403  return a.max_score > b.max_score;
404  });
405  return {essential, non_essential};
406  }
407 
410  bool update_non_essential(Partition& partition, Score available_threshold)
411  {
412  if (partition.essential[partition.essential.size() - 1].max_score
413  <= available_threshold) {
414  while (partition.essential[partition.essential.size() - 1].max_score
415  <= available_threshold) {
416  List list = partition.essential[partition.essential.size() - 1];
417  partition.non_essential.push_back(list);
418  partition.essential.pop_back();
419  available_threshold -= list.max_score;
420  }
421  return true;
422  }
423  return false;
424  }

427  const std::vector<Score>& term_weights,

461  auto curscore = iterators[post_idx].current->score;
462  score += curscore * term_weights[post_idx];

464  if (iterators[post_idx].current != iterators[post_idx].end) {
465  essential_by_doc.pop_push(

467  } else {
468  essential_by_doc.pop();
469  }
470  }

472  for (auto& list : partition.non_essential) {
473  auto& current = iterators[list.term].current;
474  current =
475  term_postings[list.term].next_ge(current, min_doc);
476  auto& [current_doc, current_score] = *current;
477  if (min_doc == current_doc) {
