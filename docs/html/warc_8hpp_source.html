83  std::istream& read_version(std::istream& in, std::string& version)
84  {
85  std::string line;
86  std::getline(in, line);
87  if (line == "" && !std::getline(in, line)) {
88  return in;
89  }
90  std::regex version_pattern("^WARC/(.+)$");
91  std::smatch sm;
92  if (!std::regex_search(line, sm, version_pattern)) {
93  throw warc_format_exception(line, "could not parse version: ");
94  }
95  version = sm.str(1);
96  return in;
97  }
98 
100  void read_fields(std::istream& in, field_map_type& fields)
101  {
102  std::string line;
103  std::getline(in, line);
104  while (line != "") {
105  std::regex field_pattern("^(.+):\\s+(.*)$");
106  std::smatch sm;
107  if (!std::regex_search(line, sm, field_pattern)) {
108  //throw WarcFormatException(line, "could not parse field: ");
109  std::cerr << "cound not parse field: " << line << std::endl;
110  } else {
111  fields[sm.str(1)] = sm.str(2);
112  }
113  std::getline(in, line);
114  }
115  }

125 std::istream& read_warc_record(std::istream& in, warc_record& record)
126 {
127  std::string version;
128  if (!warc::read_version(in, version)) {
129  return in;
130  }
131  record.version_ = std::move(version);
132  warc::read_fields(in, record.warc_fields_);
133  std::string line;
134  if (record.type() != "warcinfo") {
135  std::getline(in, line);
136  }
137  warc::read_fields(in, record.http_fields_);
138  if (record.type() != "warcinfo") {
139  try {
140  std::size_t length = std::stoi(record.content_length());
141  record.content_.resize(length);
142  in.read(&record.content_[0], length);
143  std::getline(in, line);
144  std::getline(in, line);
145  } catch (std::invalid_argument& error) {
146  throw warc_format_exception(
147  record.content_length(), "could not parse content length: ");
148  }
149  }
150  return in;
151 }
