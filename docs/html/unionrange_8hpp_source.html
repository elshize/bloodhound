147  std::vector<docterm> select_pivot(score_type threshold)
148  {
149  score_type sum_max_scores(0);
150  std::vector<docterm> preceding;
151  while (!heap_.empty()) {
152  std::pop_heap(heap_.begin(), heap_.end());
153  preceding.push_back(heap_.back());
154  auto term = preceding.back().term;
155  sum_max_scores += max_scores_[term] * weights_[term];
156  heap_.pop_back();
157  if (sum_max_scores >= threshold) {
158  break;
159  }
160  }
161  auto pivot_doc = preceding.back().doc;
162  while (!heap_.empty() && peek_posting().doc == pivot_doc) {
163  std::pop_heap(heap_.begin(), heap_.end());
164  preceding.push_back(heap_.back());
165  heap_.pop_back();
166  }
167  return preceding;
168  }

170  posting_type next_doc_wand(score_type threshold)
171  {
172  while (true) {
173  auto preceding = select_pivot(threshold);
174  auto pivot_doc = preceding.back().doc;
175  if (preceding.front().doc == pivot_doc) {
176  score_type score(0);
177  for (auto & [doc, term] : preceding) {
178  score += ranges_[term].front().score;
179  ranges_[term].advance();
180  if (!ranges_[term].empty()) {
181  heap_.push_back({doc, term});
182  std::push_heap(heap_.begin(), heap_.end());
183  }
184  }
185  return {pivot_doc, score};
186  } else {
187  // Move
188  for (auto & [doc, term] : preceding) {
189  nextge(term, pivot_doc);
190  if (!ranges_[term].empty()) {
191  heap_.push_back({doc, term});
192  std::push_heap(heap_.begin(), heap_.end());
193  }
194  }
195  }
196  }
