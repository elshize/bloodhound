56  struct doc_freq_pair {
57  document_type doc;
58  frequency_type freq;
59  bool operator==(const doc_freq_pair& rhs) const
60  {
61  return doc == rhs.doc && freq == rhs.freq;

90  void add_term(const term_type& term)
91  {
92  ++all_occurrences_;
93  ++document_sizes_[current_doc_];
94  auto ti = term_map_.find(term);
95  if (ti != term_map_.end()) {
96  term_id_type term_id = ti->second;
97  if (postings_[term_id].back().doc == current_doc_) {
98  postings_[term_id].back().freq++;
99  } else {
100  postings_[term_id].push_back({current_doc_, 1});
101  }
102  term_occurrences_[term_id]++;
103  } else {
104  term_id_type term_id = term_map_.size();
105  term_map_[term] = term_id;
106  postings_.push_back({{current_doc_, 1}});
107  term_occurrences_.push_back(1);
108  }
109  }
110 
112  frequency_type document_frequency(term_id_type term_id)
113  {
114  return postings_[term_id].size();
115  }

118  std::size_t term_count() { return term_map_.size(); }

121  std::size_t collection_size() { return current_doc_ + 1; }

124  void sort_terms()
125  {
126  sorted_terms_ = std::vector<term_type>();
127  for (const auto& term_entry : term_map_) {
128  const term_type& term = term_entry.first;
129  sorted_terms_->push_back(term);
130  }
131  std::sort(sorted_terms_->begin(), sorted_terms_->end());
132 
133  // Create new related structures and swap.
134  std::vector<std::vector<doc_freq_pair>> postings;
135  std::vector<frequency_type> term_occurrences;
136  for (const std::string& term : sorted_terms_.value()) {
137  term_id_type term_id = term_map_[term];
138  term_map_[term] = postings.size();
139  postings.push_back(std::move(postings_[term_id]));
140  term_occurrences.push_back(term_occurrences_[term_id]);
141  }
142  postings_ = std::move(postings);
143  term_occurrences_ = std::move(term_occurrences);
144  }

167  index::block_list_builder<document_type, true> list_builder(
168  block_size_, varbyte_codec<document_type>{});

171  offset += list_builder.write(out);
172  }
173  offset_table<> offset_table = build_offset_table<>(offsets);
174  off << offset_table;
