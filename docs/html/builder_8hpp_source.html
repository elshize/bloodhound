90  void add_term(const term_type& term)
91  {
92  ++all_occurrences_;
93  ++document_sizes_.back();
94  auto ti = term_map_.find(term);
95  if (ti != term_map_.end()) {
96  term_id_type term_id = ti->second;
97  if (postings_[term_id].back().doc == current_doc_) {
98  postings_[term_id].back().freq++;
99  } else {
100  postings_[term_id].push_back({current_doc_, 1});
101  }
102  term_occurrences_[term_id]++;
103  } else {
104  term_id_type term_id = term_map_.size();
105  term_map_[term] = term_id;
106  postings_.push_back({{current_doc_, 1}});
107  term_occurrences_.push_back(1);
108  }
109  }

125  void sort_terms()
126  {
127  sorted_terms_ = std::vector<term_type>();
128  for (const auto& term_entry : term_map_) {
129  const term_type& term = term_entry.first;
130  sorted_terms_->push_back(term);
131  }
132  std::sort(sorted_terms_->begin(), sorted_terms_->end());
133 
134  // Create new related structures and swap.
135  std::vector<std::vector<doc_freq_pair>> postings;
136  std::vector<frequency_type> term_occurrences;
137  for (const std::string& term : sorted_terms_.value()) {
138  term_id_type term_id = term_map_[term];
139  term_map_[term] = postings.size();
140  postings.push_back(std::move(postings_[term_id]));
141  term_occurrences.push_back(term_occurrences_[term_id]);
142  }
143  postings_ = std::move(postings);
144  term_occurrences_ = std::move(term_occurrences);
145  }
