51  struct doc_freq_pair {

76  void add_term(const term_type& term)
77  {
78  auto ti = term_map_.find(term);
79  if (ti != term_map_.end()) {
80  term_id_type term_id = ti->second;
81  if (postings_[term_id].back().doc == current_doc_) {
82  postings_[term_id].back().freq++;
83  } else {
84  postings_[term_id].push_back({current_doc_, 1});
85  }
86  term_occurrences_[term_id]++;
87  } else {
88  term_id_type term_id = term_map_.size();
89  term_map_[term] = term_id;
90  postings_.push_back({{current_doc_, 1}});
91  term_occurrences_.push_back(1);
92  }
93  }

96  frequency_type document_frequency(term_id_type term_id)
97  {
98  return postings_[term_id].size();
99  }

108  void sort_terms()
109  {
110  sorted_terms_ = std::vector<term_type>();
111  for (const auto& term_entry : term_map_) {
112  const term_type& term = term_entry.first;
113  sorted_terms_->push_back(term);
114  }
115  std::sort(sorted_terms_->begin(), sorted_terms_->end());
116 
117  // Create new related structures and swap.
118  std::vector<std::vector<doc_freq_pair>> postings;
119  std::vector<std::size_t> term_occurrences;
120  for (const std::string& term : sorted_terms_.value()) {
121  term_id_type term_id = term_map_[term];
122  term_map_[term] = postings.size();
123  postings.push_back(std::move(postings_[term_id]));
124  term_occurrences.push_back(std::move(term_occurrences_[term_id]));
125  }
126  postings_ = std::move(postings);
127  term_occurrences_ = std::move(term_occurrences);
128  }

142  void write_document_ids(std::ostream& out, std::ostream& off)
143  {
144  if (sorted_terms_ == std::nullopt) {
145  sort_terms();
146  }
147  std::size_t offset = 0;
148  //std::size_t offset_delta = 0;
149  std::vector<std::size_t> offsets;
150  for (auto& term : sorted_terms_.value()) {
151  offsets.push_back(offset);
152  term_id_type term_id = term_map_[term];
153  document_type last = 0;
154  std::vector<document_type> deltas;
155  for (const auto& posting : postings_[term_id]) {
156  deltas.push_back(posting.doc - last);
157  last = posting.doc;
158  }
159  auto encoded =
160  coding::encode<coding::varbyte_codec<frequency_type>>(deltas);
161  out.write(encoded.data(), encoded.size());
162  offset += encoded.size();
163  }
164  offset_table<> offset_table(offsets);
165  off << offset_table;
166  }

169  void write_document_counts(std::ostream& out, std::ostream& off)
170  {
171  if (sorted_terms_ == std::nullopt) {
172  sort_terms();
173  }
174  std::size_t offset = 0;
175  std::vector<std::size_t> offsets;
176  for (auto& term : sorted_terms_.value()) {
177  offsets.push_back(offset);
178  term_id_type term_id = term_map_[term];
179  auto freq_fn = [](const auto& posting) { return posting.freq; };
180  auto encoded =
181  coding::encode_fn<coding::varbyte_codec<frequency_type>>(
182  postings_[term_id], freq_fn);
183  out.write(encoded.data(), encoded.size());
184  offset += encoded.size();
185  }
186  offset_table<> offset_table(offsets);
187  off << offset_table;
188  }
189 
191  void write_document_frequencies(std::ostream& out)
192  {
193  if (sorted_terms_ == std::nullopt) {
194  sort_terms();
195  }
196  std::vector<frequency_type> dfs;
197  for (auto& term : sorted_terms_.value()) {
198  term_id_type term_id = term_map_[term];
199  dfs.push_back(document_frequency(term_id));
200  }
201  auto encoded =
202  coding::encode<coding::varbyte_codec<frequency_type>>(dfs);
203  out.write(encoded.data(), encoded.size());
204  }
