108 template<class Key, class Value>
109 std::vector<std::pair<Key, Value>>
110 aggregate_top_k(const block_accumulator_vector<Value>& accumulators, int k)
111 {
112  irk::top_k_accumulator<Key, Value> top(k);
113  for (std::size_t block = 0; block < accumulators.max_values.size(); ++block)
114  {
115  if (accumulators.max_values[block] < top.threshold()) { continue; }
116  std::size_t begin = block * accumulators.block_size;
117  std::size_t end = std::min((block + 1) * accumulators.block_size,
118  accumulators.accumulators.size());
119  for (Key idx = begin; idx < end; ++idx) {
120  top.accumulate(idx, accumulators.accumulators[idx]);
121  }
122  }
123  return top.sorted();
124 }
