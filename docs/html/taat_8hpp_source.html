146 template<class Result, class AccumulatorArray>
147 std::vector<Result> aggregate_top(std::size_t k, const AccumulatorArray& acc)
148 {
149  static_assert(std::is_convertible<pure_element_t<AccumulatorArray>,
150  score_t<Result>>::value);
151  BOOST_CONCEPT_ASSERT((concept::InputRange<AccumulatorArray>));
152  using Doc = doc_t<Result>;
153  using Score = score_t<Result>;
154  top_k_accumulator<Result> topk(k);
155  Doc doc(0);
156  for (const Score& score : acc) {
157  topk.accumulate({doc, score});
158  doc++;
159  }
160  return topk.sorted();
161 }
