94  void accumulate(Posting posting)
95  {
96  if (posting.score > threshold_) {
97  top_.push_back(posting);
98  if (top_.size() <= k_) {
99  std::push_heap(top_.begin(), top_.end(), result_order);
100  } else {
101  std::pop_heap(top_.begin(), top_.end(), result_order);
102  top_.pop_back();
103  }
104  threshold_ = top_.size() == k_ ? top_[0].score : threshold_;
105  }
106  }

114  {
115  std::vector<Posting> sorted = top_;
116  std::sort(sorted.begin(), sorted.end(), result_order);
117  return sorted;
