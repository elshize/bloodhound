65 class TopKAccumulator {

73  static bool result_order(const Posting& lhs, const Posting& rhs)
74  {
75  return lhs.score > rhs.score;
76  };
77

84  TopKAccumulator(std::size_t k) : k_(k), threshold_(0){};
85

92  void accumulate(Posting posting)
93  {
94  if (posting.score > threshold_) {
95  top_.push_back(posting);
96  if (top_.size() <= k_) {
97  std::push_heap(top_.begin(), top_.end(), result_order);
98  } else {
99  std::pop_heap(top_.begin(), top_.end(), result_order);
100  top_.pop_back();
101  }
102  threshold_ = top_.size() == k_ ? top_[0].score : threshold_;
103  }
104  }
105 
107 
111  std::vector<Posting> sorted()
112  {
113  std::vector<Posting> sorted = top_;
114  std::sort(sorted.begin(), sorted.end(), result_order);
115  return sorted;
116  }
117

124  Score threshold() { return threshold_; }
125 };
