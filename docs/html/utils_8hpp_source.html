69 class top_k_accumulator {
70 public:
71  using key_type = Key;
72  using value_type = Value;

80  {
81  return lhs.second > rhs.second;
82  };

90  explicit top_k_accumulator(std::size_t k) : k_(k){};
91

98  void accumulate(key_type key, value_type value)
99  {
100  if (value > threshold_) {
101  top_.emplace_back(key, value);
102  if (top_.size() <= k_) {
103  std::push_heap(top_.begin(), top_.end(), result_order);
104  } else {
105  std::pop_heap(top_.begin(), top_.end(), result_order);
106  top_.pop_back();
107  }
108  threshold_ = top_.size() == k_ ? top_[0].second : threshold_;
109  }
110  }
111 
113 
117  std::vector<entry_type> sorted()
118  {
119  std::vector<entry_type> sorted = top_;
120  std::sort(sorted.begin(), sorted.end(), result_order);
121  return sorted;
122  }

131 };
