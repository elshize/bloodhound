77  template<class Symbol = char>
78  void join_next_valid(std::list<node_ptr<Symbol>>& forest)
79  {
80  using node_ptr = node_ptr<Symbol>;
81  using iterator_type = typename std::list<node_ptr>::iterator;
82  //std::pair<node_ptr, node_ptr> selected =
83  // std::make_pair<node_ptr, node_ptr>(nullptr, nullptr);
84  std::pair<iterator_type, iterator_type> selected =
85  std::make_pair(forest.end(), forest.end());
86  std::size_t selected_freq = std::numeric_limits<std::size_t>::max();
87  for (auto it = forest.begin(); it != forest.end(); ++it) {
88  auto left = *it;
89  auto right_it = std::next(it);
90  while (right_it != forest.end()) {
91  auto right = *right_it;
92  auto freq = left->frequency + right->frequency;
93  if (selected_freq > freq) {
94  //selected = std::make_pair(left, right);
95  selected = std::make_pair(it, right_it);
96  selected_freq = freq;
97  }
98  if (right->symbol != std::nullopt) {
99  break;
100  }
101  ++right_it;
102  }
103  }
104  join_selected(forest, selected);
105  }

130  template<class Symbol = char>
131  std::list<level_node<Symbol>> tag_leaves(node_ptr<Symbol> root)
132  {
133  if (root->symbol.has_value()) {
134  return {{0, root}};
135  }
136  std::vector<level_node<Symbol>> leaves;
137  std::list<level_node<Symbol>> stack;
138  level_node<Symbol> current = {0, root};
139  while (true) {
140  if (current.node->symbol != std::nullopt) {
141  // Leaf.
142  leaves.push_back(current);
143  if (stack.empty()) {
144  break;
145  }
146  current = stack.back();
147  stack.pop_back();
148  } else {
149  stack.push_back({current.level + 1, current.node->right});
150  current = {current.level + 1, current.node->left};
151  }
152  }
153  std::sort(
154  leaves.begin(), leaves.end(), [](const auto& lhs, const auto& rhs) {
155  return lhs.node->symbol < rhs.node->symbol;
156  });
157  return std::list<level_node<Symbol>>(leaves.begin(), leaves.end());
158  }

161  template<class Symbol = char>
162  node_ptr<Symbol> reconstruct(std::list<level_node<Symbol>>& nodes)
163  {
164  EXPECTS(nodes.size() > 1);
165  std::list<level_node<Symbol>> stack;
166  std::list<Symbol> propagated_symbols;
167  while (true) {
168  if (stack.size() > 1
169  && stack.back().level == std::next(stack.rbegin())->level) {
170  auto right = stack.back();
171  stack.pop_back();
172  auto left = stack.back();
173  stack.pop_back();
174  Symbol rs = propagated_symbols.back();
175  propagated_symbols.pop_back();
176  Symbol ls = propagated_symbols.back();
177  propagated_symbols.pop_back();
178  level_node<Symbol> joined = {left.level - 1,
179  join_nodes_with_symbol(left.node, right.node, ls)};
180  if (joined.level == 0) {
181  return joined.node;
182  }
183  stack.push_back(joined);
184  propagated_symbols.push_back(rs);
185  } else if (not nodes.empty()) {
186  auto next = nodes.front();
187  nodes.pop_front();
188  stack.push_back(next);
189  propagated_symbols.push_back(next.node->symbol.value());
190  } else {
191  throw std::runtime_error(
192  "wrong level alignment: check first-phase");
193  }
194  }
195  }

198  //template<class Symbol = char>
199  inline alphabetical_bst<char, uint16_t, std::vector<char>>
200  compact(const node_ptr<char>& root)
201  {
202  using alphabetical_bst =
203  alphabetical_bst<char, uint16_t, std::vector<char>>;
204  using node = typename alphabetical_bst::node;
205  std::size_t node_size = alphabetical_bst::node_size;
206 
207  std::vector<node> compact_nodes;
208  std::list<node_ptr<char>> queue;
209  queue.push_back(root);
210 
211  auto is_leaf = [](node_ptr<char> n) { return n->left == nullptr; };
212  auto calc_ptr = [&compact_nodes, &queue, node_size, is_leaf](auto n) {
213  uint16_t ptr;
214  if (is_leaf(n)) {
215  ptr = static_cast<uint16_t>(
216  static_cast<unsigned char>(n->symbol.value()));
217  } else {
218  ptr = alphabetical_bst::symbol_bound
219  + (compact_nodes.size() + queue.size() + 1) * node_size;
220  queue.push_back(n);
221  }
222  return ptr;
223  };
224 
225  while (not queue.empty())
226  {
227  auto n = queue.front();
228  queue.pop_front();
229  auto left_ptr = calc_ptr(n->left);
230  auto right_ptr = calc_ptr(n->right);
231  compact_nodes.emplace_back(n->symbol.value(), left_ptr, right_ptr);
232  }
233 
234  std::vector<char> mem;
235  for (const auto& node : compact_nodes) {
236  mem.insert(mem.end(), std::begin(node.bytes), std::end(node.bytes));
237  }
238  return alphabetical_bst(mem);
239  }
