75  template<class Symbol = char>
76  void join_next_valid(std::list<node_ptr<Symbol>>& forest)
77  {
78  using node_ptr = node_ptr<Symbol>;
79  using iterator_type = typename std::list<node_ptr>::iterator;
80  //std::pair<node_ptr, node_ptr> selected =
81  // std::make_pair<node_ptr, node_ptr>(nullptr, nullptr);
82  std::pair<iterator_type, iterator_type> selected =
83  std::make_pair(forest.end(), forest.end());
84  std::size_t selected_freq = std::numeric_limits<std::size_t>::max();
85  for (auto it = forest.begin(); it != forest.end(); ++it) {
86  auto left = *it;
87  auto right_it = std::next(it);
88  while (right_it != forest.end()) {
89  auto right = *right_it;
90  auto freq = left->frequency + right->frequency;
91  if (selected_freq > freq) {
92  //selected = std::make_pair(left, right);
93  selected = std::make_pair(it, right_it);
94  selected_freq = freq;
95  }
96  if (right->symbol != std::nullopt) {
97  break;
98  }
99  ++right_it;
100  }
101  }
102  join_selected(forest, selected);
103  }

128  template<class Symbol = char>
129  std::list<level_node<Symbol>> tag_leaves(node_ptr<Symbol> root)
130  {
131  if (root->symbol.has_value()) {
132  return {{0, root}};
133  }
134  std::vector<level_node<Symbol>> leaves;
135  std::list<level_node<Symbol>> stack;
136  level_node<Symbol> current = {0, root};
137  while (true) {
138  if (current.node->symbol != std::nullopt) {
139  // Leaf.
140  leaves.push_back(current);
141  if (stack.empty()) {
142  break;
143  }
144  current = stack.back();
145  stack.pop_back();
146  } else {
147  stack.push_back({current.level + 1, current.node->right});
148  current = {current.level + 1, current.node->left};
149  }
150  }
151  std::sort(
152  leaves.begin(), leaves.end(), [](const auto& lhs, const auto& rhs) {
153  return lhs.node->symbol < rhs.node->symbol;
154  });
155  return std::list<level_node<Symbol>>(leaves.begin(), leaves.end());
156  }

159  template<class Symbol = char>
160  node_ptr<Symbol> reconstruct(std::list<level_node<Symbol>>& nodes)
161  {
162  Expects(nodes.size() > 1);
163  std::list<level_node<Symbol>> stack;
164  std::list<Symbol> propagated_symbols;
165  while (true) {
166  if (stack.size() > 1
167  && stack.back().level == std::next(stack.rbegin())->level) {
168  auto right = stack.back();
169  stack.pop_back();
170  auto left = stack.back();
171  stack.pop_back();
172  Symbol rs = propagated_symbols.back();
173  propagated_symbols.pop_back();
174  Symbol ls = propagated_symbols.back();
175  propagated_symbols.pop_back();
176  level_node<Symbol> joined = {left.level - 1,
177  join_nodes_with_symbol(left.node, right.node, ls)};
178  if (joined.level == 0) {
179  return joined.node;
180  }
181  stack.push_back(joined);
182  propagated_symbols.push_back(rs);
183  } else if (nodes.size() > 0) {
184  auto next = nodes.front();
185  nodes.pop_front();
186  stack.push_back(next);
187  propagated_symbols.push_back(next.node->symbol.value());
188  } else {
189  throw std::runtime_error(
190  "wrong level alignment: check first-phase");
191  }
192  }
193  }

197  alphabetical_bst<char, uint16_t, std::vector<char>>
198  compact(node_ptr<char> root)
199  {
200  using alphabetical_bst =
201  alphabetical_bst<char, uint16_t, std::vector<char>>;
202  using node = typename alphabetical_bst::node;
203  std::size_t node_size = alphabetical_bst::node_size;
204 
205  std::vector<node> compact_nodes;
206  std::list<node_ptr<char>> queue;
207  queue.push_back(root);
208 
209  auto is_leaf = [](node_ptr<char> n) { return n->left == nullptr; };
210  auto calc_ptr = [&compact_nodes, &queue, node_size, is_leaf](auto n) {
211  uint16_t ptr;
212  if (is_leaf(n)) {
213  ptr = static_cast<uint16_t>((unsigned char)n->symbol.value());
214  } else {
215  ptr = alphabetical_bst::symbol_bound
216  + (compact_nodes.size() + queue.size() + 1) * node_size;
217  queue.push_back(n);
218  }
219  return ptr;
220  };
221 
222  while (!queue.empty()) {
223  auto n = queue.front();
224  queue.pop_front();
225  auto left_ptr = calc_ptr(n->left);
226  auto right_ptr = calc_ptr(n->right);
227  compact_nodes.push_back(
228  node(n->symbol.value(), left_ptr, right_ptr));
229  }
230 
231  std::vector<char> mem;
232  for (const auto& node : compact_nodes) {
233  mem.insert(mem.end(), node.bytes, node.bytes + node_size);
234  }
235  return alphabetical_bst(mem);
236  }
