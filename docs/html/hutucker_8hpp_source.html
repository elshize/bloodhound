71  template<class Symbol = char>
72  void join_next_valid(std::list<node_ptr<Symbol>>& forest)
73  {
74  using node_ptr = node_ptr<Symbol>;
75  std::pair<node_ptr, node_ptr> selected =
76  std::make_pair<node_ptr, node_ptr>(nullptr, nullptr);
77  std::size_t selected_freq;
78  for (auto it = forest.begin(); it != forest.end(); ++it) {
79  auto left = *it;
80  auto right_it = std::next(it);
81  while (right_it != forest.end()) {
82  auto right = *right_it++;
83  auto freq = left->frequency + right->frequency;
84  if (selected.first == nullptr || selected_freq > freq) {
85  selected = std::make_pair(left, right);
86  selected_freq = freq;
87  }
88  if (right->symbol != std::nullopt) {
89  break;
90  }
91  }
92  }
93  join_selected(forest, selected);
94  }

116  template<class Symbol = char>
117  std::list<level_node<Symbol>> tag_leaves(node_ptr<Symbol> root)
118  {
119  if (root->symbol != std::nullopt) {
120  return {{0, root}};
121  }
122  std::vector<level_node<Symbol>> leaves;
123  std::list<level_node<Symbol>> stack;
124  level_node<Symbol> current = {0, root};
125  while (true) {
126  if (current.node->symbol != std::nullopt) {
127  // Leaf.
128  leaves.push_back(current);
129  if (stack.empty()) {
130  break;
131  }
132  current = stack.back();
133  stack.pop_back();
134  } else {
135  stack.push_back({current.level + 1, current.node->right});
136  current = {current.level + 1, current.node->left};
137  }
138  }
139  std::sort(
140  leaves.begin(), leaves.end(), [](const auto& lhs, const auto& rhs) {
141  return lhs.node->symbol < rhs.node->symbol;
142  });
143  return std::list<level_node<Symbol>>(leaves.begin(), leaves.end());
144  }

147  template<class Symbol = char>
148  node_ptr<Symbol> reconstruct(std::list<level_node<Symbol>>& nodes)
149  {
150  std::list<level_node<Symbol>> stack;
151  std::list<Symbol> propagated_symbols;
152  while (true) {
153  if (stack.size() > 1
154  && stack.back().level == std::next(stack.rbegin())->level) {
155  auto right = stack.back();
156  stack.pop_back();
157  auto left = stack.back();
158  stack.pop_back();
159  Symbol rs = propagated_symbols.back();
160  propagated_symbols.pop_back();
161  Symbol ls = propagated_symbols.back();
162  propagated_symbols.pop_back();
163  level_node<Symbol> joined = {left.level - 1,
164  join_nodes_with_symbol(left.node, right.node, ls)};
165  if (joined.level == 0) {
166  return joined.node;
167  }
168  stack.push_back(joined);
169  propagated_symbols.push_back(rs);
170  } else {
171  auto next = nodes.front();
172  nodes.pop_front();
173  stack.push_back(next);
174  propagated_symbols.push_back(next.node->symbol.value());
175  }
176  }
177  }

180  template<class Symbol = char>
181  alphabetical_bst<Symbol, uint16_t, std::vector<char>>
182  compact(node_ptr<Symbol> root)
183  {
184  using alphabetical_bst =
185  alphabetical_bst<Symbol, uint16_t, std::vector<char>>;
186  using node = typename alphabetical_bst::node;
187  std::size_t node_size = alphabetical_bst::node_size;
188 
189  std::vector<node> compact_nodes;
190  std::list<node_ptr<Symbol>> queue;
191  queue.push_back(root);
192 
193  auto is_leaf = [](node_ptr<Symbol> n) { return n->left == nullptr; };
194  auto calc_ptr = [&compact_nodes, &queue, node_size, is_leaf](auto n) {
195  uint16_t ptr;
196  if (is_leaf(n)) {
197  ptr = static_cast<uint16_t>(n->symbol.value());
198  } else {
199  ptr = alphabetical_bst::symbol_bound
200  + (compact_nodes.size() + queue.size() + 1) * node_size;
201  queue.push_back(n);
202  }
203  return ptr;
204  };
205 
206  while (!queue.empty()) {
207  auto n = queue.front();
208  queue.pop_front();
209  auto left_ptr = calc_ptr(n->left);
210  auto right_ptr = calc_ptr(n->right);
211  compact_nodes.push_back(
212  node(n->symbol.value(), left_ptr, right_ptr));
213  }
214 
215  std::vector<char> mem;
216  for (const auto& node : compact_nodes) {
217  mem.insert(mem.end(), node.bytes, node.bytes + node_size);
218  }
219  return alphabetical_bst(mem);
220  }
