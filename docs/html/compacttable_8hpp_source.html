66 template<class Mem, class Codec>
67 auto read_compact_value(Mem mem, std::uint32_t key, Codec codec)
68 {
69  auto header = reinterpret_cast<const compact_table_header*>(mem.data());
70  bool delta_encoded = header->flags & CompactTableHeaderFlags::DeltaEncoding;
71  auto leader_count =
72  (header->count + header->block_size - 1) / header->block_size;
73  gsl::span<const compact_table_leader> leaders(
74  reinterpret_cast<const compact_table_leader*>(
75  mem.data() + sizeof(*header)),
76  leader_count);
77  auto leader = std::lower_bound(
78  leaders.begin(), leaders.end(), compact_table_leader{key, 0});
79  if (leader == leaders.end() || leader->key > key) {
80  // Legal because the first leader key must be 0.
81  leader--;
82  }
83  const char* block_beg = mem.data() + leader->ptr;
84  std::size_t num_skip = key - leader->key;
85  auto decoded = delta_encoded ? delta_decode(codec, block_beg, num_skip + 1)
86  : decode(codec, block_beg, num_skip + 1);
87  return decoded.back();
88 }

185 template<class T, class Codec = vbyte_codec<T>>
186 compact_table<T, Codec, std::vector<char>>
187 build_compact_table(const std::vector<T>& values,
188  bool delta_encoded = false,
189  std::uint32_t block_size = 256)
190 {
191  std::vector<char> data;
192  Codec codec;
193  auto flags = delta_encoded ? CompactTableHeaderFlags::DeltaEncoding
194  : CompactTableHeaderFlags::Default;
195  compact_table_header header{
196  static_cast<std::uint32_t>(values.size()), block_size, flags};
197  io::append_object(header, data);
198 
199  std::uint32_t block_count =
200  (header.count + header.block_size - 1) / header.block_size;
201  std::uint32_t data_offset =
202  sizeof(header) + block_count * sizeof(compact_table_leader);
203 
204  std::vector<char> blocks;
205  std::vector<compact_table_leader> leaders;
206  for (std::size_t block = 0; block < block_count; ++block) {
207  std::uint32_t beg = block * block_size;
208  std::uint32_t end = std::min(
209  beg + block_size, static_cast<std::uint32_t>(values.size()));
210  leaders.push_back(
211  {beg, data_offset + static_cast<std::uint32_t>(blocks.size())});
212  gsl::span<const T> block_span(values.data() + beg, values.data() + end);
213  auto encoded_block = delta_encoded ? delta_encode(codec, block_span)
214  : encode(codec, block_span);
215  blocks.insert(blocks.end(), encoded_block.begin(), encoded_block.end());
216  }
217  io::append_collection(leaders, data);
218  data.insert(data.end(), blocks.begin(), blocks.end());
219  return compact_table<T, Codec>(std::move(data));
220 }
