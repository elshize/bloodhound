60 template<class Mem, class Codec>
61 std::size_t read_compact_value(Mem mem, std::uint32_t key, Codec codec)
62 {
63  auto header = reinterpret_cast<const compact_table_header*>(mem.data());
64  auto leader_count =
65  (header->count + header->block_size - 1) / header->block_size;
66  gsl::span<const compact_table_leader> leaders(
67  reinterpret_cast<const compact_table_leader*>(
68  mem.data() + sizeof(*header)),
69  leader_count);
70  auto leader = std::lower_bound(
71  leaders.begin(), leaders.end(), compact_table_leader{key, 0});
72  if (leader == leaders.end() || leader->key > key) {
73  // Legal because the first leader key must be 0.
74  leader--;
75  }
76  const char* block_beg = mem.data() + leader->ptr;
77  std::size_t num_skip = key - leader->key;
78  auto decoded = coding::decode_delta_n(
79  gsl::span<const char>(block_beg, mem.size() - leader->ptr),
80  num_skip + 1,
81  codec);
82  return decoded.back();
83 }

105 template<class T, class Codec = coding::varbyte_codec<T>>
106 class compact_table {
107  static_assert(std::is_convertible<T, typename Codec::value_type>::value);
108 
109 public:
110  using value_type = typename Codec::value_type;
111 
112 protected:
113  Codec codec_;
114  std::vector<char> data_;
115 
116 public:
117  compact_table(fs::path file) { io::load_data(file, data_); }
118  compact_table(const std::vector<T>& values,
119  bool delta_encoded = false,
120  std::uint32_t block_size = 256)
121  {
122  auto flags = delta_encoded ? CompactTableHeaderFlags::DeltaEncoding
123  : CompactTableHeaderFlags::Default;
124  compact_table_header header{
125  static_cast<std::uint32_t>(values.size()), block_size, flags};
126  io::append_object(header, data_);
127 
128  std::uint32_t block_count =
129  (header.count + header.block_size - 1) / header.block_size;
130  std::uint32_t data_offset =
131  sizeof(header) + block_count * sizeof(compact_table_leader);
132 
133  std::vector<char> blocks;
134  std::vector<compact_table_leader> leaders;
135  for (std::size_t block = 0; block < block_count; ++block) {
136  std::uint32_t beg = block * block_size;
137  std::uint32_t end = std::min(
138  static_cast<std::size_t>(beg + block_size), values.size());
139  leaders.push_back(
140  {beg, data_offset + static_cast<std::uint32_t>(blocks.size())});
141  gsl::span<const T> block_span(
142  values.data() + beg, values.data() + end);
143  auto encoded_block = delta_encoded
144  ? coding::encode_delta(block_span, codec_)
145  : coding::encode(block_span, codec_);
146  blocks.insert(
147  blocks.end(), encoded_block.begin(), encoded_block.end());
148  }
149  io::append_collection(leaders, data_);
150  data_.insert(data_.end(), blocks.begin(), blocks.end());
151  }
152 
153  std::size_t operator[](std::size_t term_id)
154  {
155  return read_compact_value(data_, term_id, codec_);
156  }
157 
158  std::size_t operator[](std::size_t term_id) const
159  {
160  return read_compact_value(data_, term_id, codec_);
161  }
162 
163  const char* data() { return data_.data(); }
164 
165  const compact_table_header* header() const
166  {
167  return reinterpret_cast<const compact_table_header*>(data_.data());
168  };
169 
170  std::size_t size() const { return header()->count; }
171 
172  bool operator==(const compact_table<T, Codec>& rhs)
173  {
174  return data_ == rhs.data_;
175  }
176 
177  friend std::ostream&
178  operator<<<>(std::ostream&, const compact_table<T, Codec>&);
179 };
