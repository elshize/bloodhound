89  void insert_node(
90  node_ptr n, const bitword& encoded, std::size_t bitn, Value value)
91  {
92  bool first_bit = encoded.test(bitn++);
93  auto bits = encoded >> bitn;
94  bits.resize(encoded.size() - bitn);
95  node_ptr new_node = std::make_shared<node>(
96  bits, nullptr, nullptr, std::make_optional(value));
97  if (first_bit) {
98  n->right = new_node;
99  } else {
100  n->left = new_node;
101  }
102  }

104  void break_node(node_ptr n, bitword&& prefix)
105  {
106  bool right = n->bits[prefix.size()];
107  bitword suffix = n->bits >> (prefix.size() + 1);
108  suffix.resize(n->bits.size() - prefix.size() - 1);
109  n->bits = prefix;
110  node_ptr new_node = std::make_shared<node>(
111  suffix, n->left, n->right, n->value);
112  if (right) {
113  n->right = new_node;
114  n->left = nullptr;
115  } else {
116  n->left = new_node;
117  n->right = nullptr;
118  }
119  n->value = std::nullopt;
120  n->sentinel = false;
121  }

131  std::pair<std::size_t, bitword> common(std::size_t bitpos,
132  const bitword& inserted,
133  const bitword& internal) const
134  {
135  bitword common;
136  while (common.size() < internal.size() && bitpos < inserted.size()
137  && inserted[bitpos] == internal[common.size()]) {
138  common.push_back(inserted[bitpos++]);
139  }
140  return std::make_pair(bitpos, std::move(common));
141  }

165  search_result search(const bitword& encoded, std::size_t begin = 0) const
166  {
167  node_ptr current_node = root_;
168  node_ptr last_left = nullptr;
169  std::size_t pos = begin;
170  while (pos < encoded.size()) {
171  std::size_t node_beg = pos;
172  std::size_t bidx = 0;
173  for (; bidx < current_node->bits.size() && pos < encoded.size();
174  ++bidx, ++pos) {
175  if (encoded[pos] != current_node->bits[bidx]) {
176  return {false, node_beg, pos, current_node, last_left};
177  }
178  }
179  if (pos == encoded.size()) {
180  if (bidx == current_node->bits.size()) {
181  return {true, node_beg, pos, current_node, last_left};
182  } else {
183  return {false, node_beg, pos, current_node, last_left};
184  }
185  }
186  node_ptr child =
187  encoded[pos] ? current_node->right : current_node->left;
188  if (encoded[pos]) {
189  if (current_node->sentinel) {
190  last_left = current_node;
191  } else if (current_node->left != nullptr) {
192  last_left = current_node->left;
193  }
194  }
195  if (child != nullptr) {
196  current_node = child;
197  ++pos;
198  } else {
199  return {false, node_beg, pos, current_node, last_left};
200  }
201  }
202  return {current_node->bits.size() == 0,
203  encoded.size(),
204  encoded.size(),
205  current_node,
206  last_left};
207  }

264  bool
265  insert(const bitword& encoded, Value value)
266  {
267  std::size_t bitn = 0;
268  if (encoded.size() == 0) {
269  return false;
270  }
271  auto[pos, insertion_node] = find(encoded);
272  bitn = pos;
273  bool found_as_external = bitn == encoded.size();
274  if (found_as_external) {
275  return make_external(insertion_node, value);
276  } else {
277  if (!insertion_node->compressed()) {
278  insert_node(insertion_node, encoded, bitn, value);
279  return true;
280  } else {
281  auto[bitpos, common_prefix] =
282  common(bitn, encoded, insertion_node->bits);
283  bitn = bitpos;
284  if (common_prefix.size() < insertion_node->bits.size()) {
285  break_node(insertion_node, std::move(common_prefix));
286  }
287  if (bitn < encoded.size()) {
288  insert_node(insertion_node, encoded, bitn, value);
289  return true;
290  } else {
291  bool inserted = !insertion_node->sentinel;
292  insertion_node->sentinel = true;
293  insertion_node->value = value;
294  return inserted;
295  }
296  }
297  }

398 {

401  varbyte_codec<Value> value_codec;
402  varbyte_codec<std::size_t> size_codec;
403  std::size_t size;
404  in.read(reinterpret_cast<char*>(&size), sizeof(size));
405  for (std::size_t idx = 0; idx < size; ++idx) {
406  Value v;
407  value_codec.decode(in, v);
