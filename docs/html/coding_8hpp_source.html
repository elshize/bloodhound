151 template<class Codec, class InputRange>
152 std::ostream& encode_delta(const InputRange& int_range,
153  std::ostream& sink,
154  const Codec& codec = Codec(),
155  typename Codec::value_type initial_value = typename Codec::value_type(0))
156 {
157  BOOST_CONCEPT_ASSERT((concept::InputRange<InputRange>));
158  typename Codec::value_type prev = initial_value;
159  for (const auto& n : int_range) {
160  codec.encode(n - prev, sink);
161  prev = n;
162  }
163  return sink;
164 }

196 std::istream& decode(
197  OutputIterator output, std::istream& source, const Codec& codec = Codec())
198 {
199  BOOST_CONCEPT_ASSERT(
200  (boost::OutputIterator<OutputIterator, typename Codec::value_type>));

202  while (codec.decode(source, integer)) {
203  *output++ = integer;
