281 std::istream& decode(
282  OutputIterator output, std::istream& source, const Codec& codec = Codec())
283 {
284  BOOST_CONCEPT_ASSERT(

496 template<class SizeCodec>
497 std::ostream& encode_bits(const bitword& bits,
498  std::ostream& out,
499  const SizeCodec& size_codec = SizeCodec())
500 {
501  std::vector<bitword::block_type> bytes;
502  boost::to_block_range(bits, std::back_inserter(bytes));
503  std::size_t bits_num = bits.size();
504  size_codec.encode(bits_num, out);
505  out.write(reinterpret_cast<char*>(bytes.data()), (bits_num + 7) / 8);
506  return out;
507 }

509 template<class SizeCodec>
510 bitword decode_bits(std::istream& in, const SizeCodec& size_codec = SizeCodec())
511 {
512  bitword bits;
513  std::size_t bits_num;
514  size_codec.decode(in, bits_num);
515  std::size_t bytes_num = (bits_num + 7) / 8;
516  std::vector<char> bytes(bytes_num);
517  in.read(bytes.data(), bytes_num);
518  bitptr<char> p(bytes.data());
519  for (std::size_t bitn = 0; bitn < bits_num; ++bitn) {
520  bits.push_back(p[bitn]);
521  }
522  return bits;
523 }
