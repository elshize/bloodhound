66 inline void
67 load_data(const fs::path& data_file, std::vector<char>& data_container)
68 {
69  enforce_exist(data_file);
70  std::ifstream in(data_file.c_str(), std::ios::binary);
71  in.seekg(0, std::ios::end);
72  std::streamsize size = in.tellg();
73  in.seekg(0, std::ios::beg);
74  data_container.resize(size);
75  if (not in.read(data_container.data(), size)) {
76  throw std::runtime_error("Failed reading " + data_file.string());
77  }
78 }

80 inline void

85  std::string line;
