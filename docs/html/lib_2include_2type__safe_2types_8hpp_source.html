24  template <typename T, typename U, U Value>
25  constexpr T validate_value()
26  {
27  static_assert(sizeof(T) <= sizeof(U)
28  && std::is_signed<U>::value == std::is_signed<T>::value,
29  "mismatched types");
30  static_assert(U(std::numeric_limits<T>::min()) <= Value
31  && Value <= U(std::numeric_limits<T>::max()),
32  "integer literal overflow");
33  return static_cast<T>(Value);
34  }
