56  self_type& moveto(value_type id)
57  {
58  static_assert(
59  sorted == true, "moveto not supported for unsorted lists");
60  int block = nextgeq_block(id);
61  if (block >= view_.num_blocks())
62  {
63  finish();
64  return *this;
65  }
66  pos_ = block == block_ ? pos_ : 0;
67  block_ = block;
68  while (view_[absolute_position()] < id) { pos_++; }
69  return *this;
70  };

126  int nextgeq_block(value_type id) const
127  {
128  int block = block_;
129  while (block < view_.num_blocks() && last_in_block(block) < id)

135  {
136  auto idx = block < view_.num_blocks() - 1
137  ? view_.block_size() - 1

139  return view_[absolute_position(block, idx)];
140  }
