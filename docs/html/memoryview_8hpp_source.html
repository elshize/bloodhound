318  }

320  std::streamsize size() const { return size_; }

323 
324  disk_memory_source<char_type>
325  range(std::streamsize first, std::streamsize size) const
326  {
327  return disk_memory_source(file_path_, offset_ + first, size);
328  }
329 
330 private:
331  boost::filesystem::path file_path_{};
332  std::streamsize offset_ = 0;
333  std::streamsize size_ = 0;
334  mutable std::vector<char_type> buffer_;
335 
336  void ensure_loaded() const
337  {
338  if (buffer_.empty()) {
339  io::enforce_exist(file_path_);
340  auto flags = std::ifstream::ate | std::ifstream::binary;
341  std::ifstream in(file_path_.c_str(), flags);
342  in.seekg(offset_, std::ios::beg);
343  buffer_.resize(size_);
344  if (not in.read(buffer_.data(), size_))
345  {
346  throw std::runtime_error(
347  "Failed reading " + file_path_.string());
348  }
349  }
350  }
351 
352  std::streamsize calc_size() const
353  {
354  auto flags = std::ifstream::ate | std::ifstream::binary;
355  std::ifstream in(file_path_.c_str(), flags);
356  return in.tellg();
357  }
