82 struct bm25_scorer {
83  using tag_type = bm25_tag;
84  tag_type scoring_tag;
85  double x, y, z;
86 
87  bm25_scorer(int32_t documents_with_term_count,
88  int32_t total_document_count,
89  double avg_document_size,
90  double k1 = 1.2,
91  double b = 0.5,
92  double min_idf = 1.0E-6)
93  {
94  double idf_numerator = total_document_count - documents_with_term_count
95  + 0.5;
96  double idf_denominator = documents_with_term_count + 0.5;
97  double idf = std::max(
98  min_idf, std::log(idf_numerator / idf_denominator));
99  x = idf * (k1 + 1);
100  y = k1 - b * k1;
101  z = b * k1 / avg_document_size;
102  }
103 
105  double operator()(int32_t tf, int32_t document_size) const
106  {
107  return (tf * x) / (tf + y + (z * document_size));
108  }
109 };
