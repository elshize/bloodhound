81 struct bm25_scorer {
82  using tag_type = bm25_tag;
83  tag_type scoring_tag;

85 
86  bm25_scorer(long term_document_count,
87  long collection_document_count,
88  double avg_document_size,
89  double k1 = 1.2,
90  double b = 0.5)
91  {
92  double numerator = collection_document_count - term_document_count
93  + 0.5;
94  double denominator = term_document_count + 0.5;
95  double idf = std::log(numerator / denominator);
96  x = idf * (k1 + 1);
97  y = k1 - b * k1;
98  z = b * k1 / avg_document_size;
99  }
100 
101 
103  double operator()(long tf, long document_size) const
104  {
105  return tf;
106  }
