97  bool compressed() const { return ((*mem_) & 0b10000000) > 0; }
98  bool sentinel() const { return ((*mem_) & 0b01000000) > 0; }
99  bool has_next() const { return ((*mem_) & 0b00100000) > 0; }
100  bool has_left() const { return has_next(); }

102  constexpr std::uint32_t value_size() const
103  {

105  return 0;
106  } else {

108  }

113  boost::dynamic_bitset<unsigned char> compressed_bits() const
114  {
115  std::size_t len = compressed_len();
116  std::size_t bytes = (len + 7) / 8;
117  auto bits = boost::dynamic_bitset<unsigned char>(
118  mem_ + header_size() + value_size(),
119  mem_ + header_size() + value_size() + bytes);
120  bits.resize(len);
121  return bits;

163  Node(bool has_left, std::size_t rightptr, bool sentinel)
164  {
165  assert(rightptr < 32);
166  unsigned char header = 0;
167  header |= sentinel << 6;
168  header |= has_left << 5;
169  header |= (rightptr > 0) << 4;
170  std::uint32_t rptrlen = irk::nbytes(rightptr);
171  header |= rptrlen;
172  mem_.push_back(header);
173  std::vector<unsigned char> rightptr_data(rptrlen, 0);
174  std::memcpy(rightptr_data.data(), &rightptr, rptrlen);
175  mem_.insert(mem_.end(), rightptr_data.begin(), rightptr_data.end());
176  }
