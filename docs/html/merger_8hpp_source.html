34 template<class Doc = std::size_t,
35  class Term = std::string,
36  class TermId = std::size_t,
37  class Freq = std::size_t>
38 class index_merger {
39 public:
40  using document_type = Doc;
41  using term_type = Term;
42  using term_id_type = TermId;
43  using frequency_type = Freq;
44  using index_type =
45  inverted_index<document_type, term_type, term_id_type, frequency_type>;
46 
47 private:
48  struct entry {
49  term_id_type current_term_id;
50  const index_type* index;
51  document_type shift;
52  std::string current_term() const
53  {
54  return index->term(current_term_id);
55  }
56  bool operator<(const entry& rhs) const
57  {
58  return current_term() > rhs.current_term();
59  }
60  bool operator>(const entry& rhs) const
61  {
62  return current_term() < rhs.current_term();
63  }
64  bool operator==(const entry& rhs) const
65  {
66  return current_term() == rhs.current_term();
67  }
68  bool operator!=(const entry& rhs) const
69  {
70  return current_term() != rhs.current_term();
71  }
72  };
73  fs::path target_dir_;
74  bool skip_unique_;
75  std::vector<index_type> indices_;
76  std::vector<entry> heap_;
77  std::ofstream terms_out_;
78  std::ofstream doc_ids_;
79  std::ofstream doc_counts_;
80  std::vector<std::size_t> doc_ids_off_;
81  std::vector<std::size_t> doc_counts_off_;
82  std::vector<frequency_type> term_dfs_;
83  std::size_t doc_offset_;
84  std::size_t count_offset_;
85 
86  std::vector<entry> indices_with_next_term()
87  {
88  std::vector<entry> result;
89  std::string next_term = heap_.front().current_term();
90  while (!heap_.empty() && heap_.front().current_term() == next_term) {
91  std::pop_heap(heap_.begin(), heap_.end());
92  result.push_back(heap_.back());
93  heap_.pop_back();
94  }
95  return result;
96  }

98 public:
99  index_merger(fs::path target_dir,
100  std::vector<fs::path> indices,
101  bool skip_unique = false)
102  : target_dir_(target_dir), skip_unique_(skip_unique)
103  {
104  for (fs::path index_dir : indices) {
105  std::cout << "Loading index " << index_dir << std::endl;
106  indices_.emplace_back(index_dir, false, true);
107  }
108  terms_out_.open(index::terms_path(target_dir));
109  doc_ids_.open(index::doc_ids_path(target_dir));
110  doc_counts_.open(index::doc_counts_path(target_dir));
111  doc_offset_ = 0;
112  count_offset_ = 0;
113  }
114 
115  void merge_term(std::vector<entry>& indices)
116  {
117  if (skip_unique_ && indices.size() == 1) {
118  auto pr = indices.front().index->posting_range(
119  indices.front().current_term_id, score::count_scorer{});
120  if (pr.size() == 1) {
121  std::cerr << "Skipping unique term "
122  << indices.front().current_term() << std::endl;
123  return;
124  }
125  }
126  std::cerr << "Merging term " << term_dfs_.size() << " ("
127  << indices.front().current_term() << ")" << std::endl;
128 
129  // Write the term.
130  terms_out_ << indices.front().current_term() << std::endl;
131 
132  // Sort by shift, i.e., effectively document IDs.
133  std::sort(indices.begin(),
134  indices.end(),
135  [](const entry& lhs, const entry& rhs) {
136  return lhs.shift < rhs.shift;
137  });
138 
139  // Merge the posting lists.
140  std::vector<document_type> doc_ids;
141  std::vector<frequency_type> doc_counts;
142  for (const entry& e : indices) {
143  auto pr = e.index->posting_range(
144  e.current_term_id, score::count_scorer{});
145  for (auto p : pr) {
146  doc_ids.push_back(p.doc + e.shift);
147  doc_counts.push_back(p.score);
148  }
149  }
150 
151  // Accumulate the term's document frequency.
152  term_dfs_.push_back(doc_ids.size());
153 
154  // Accumulate offsets.
155  doc_ids_off_.push_back(doc_offset_);
156  doc_counts_off_.push_back(count_offset_);
157 
158  // Write documents and counts.
159  auto encoded_ids =
160  irk::coding::encode_delta<varbyte_codec<document_type>>(doc_ids);
161  doc_ids_.write(encoded_ids.data(), encoded_ids.size());
162  auto encoded_counts =
163  irk::coding::encode<varbyte_codec<frequency_type>>(doc_counts);
164  doc_counts_.write(encoded_counts.data(), encoded_counts.size());
165 
166  // Calc new offsets.
167  doc_offset_ += encoded_ids.size();
168  count_offset_ += encoded_counts.size();
169  }
170 
171  void merge_terms()
172  {
173  // Initialize heap: everyone starts with the first term.
174  document_type shift(0);
175  for (const index_type& index : indices_) {
176  heap_.push_back({0, &index, shift});
177  shift += index.collection_size();
178  }
179 
180  while (!heap_.empty()) {
181  std::vector<entry> indices_to_merge = indices_with_next_term();
182  merge_term(indices_to_merge);
183  for (const entry& e : indices_to_merge) {
184  if (std::size_t(e.current_term_id + 1)
185  < e.index->term_count()) {
186  heap_.push_back(
187  entry{e.current_term_id + 1, e.index, e.shift});
188  std::push_heap(heap_.begin(), heap_.end());
189  }
190  }
191  }
192 
193  // Write offsets
194  io::dump(
195  offset_table<>(doc_ids_off_), index::doc_ids_off_path(target_dir_));
196  io::dump(offset_table<>(doc_counts_off_),
197  index::doc_counts_off_path(target_dir_));
198 
199  // Write term dfs.
200  std::vector<char> encoded_dfs =
201  irk::coding::encode_delta<varbyte_codec<frequency_type>>(term_dfs_);
202  std::ofstream dfs(index::term_doc_freq_path(target_dir_));
203  dfs.write(encoded_dfs.data(), encoded_dfs.size());
204  dfs.close();
205 
206  // Close other streams.
207  terms_out_.close();
208  doc_ids_.close();
209  doc_counts_.close();
210  }
211 
212  void merge_titles()
213  {
214  std::ofstream tout(index::titles_path(target_dir_));
215  for (const auto& index : indices_) {
216  for (const auto& title : index.titles()) {
217  tout << title << std::endl;
218  }
219  }
220  tout.close();
221  }
222 };
