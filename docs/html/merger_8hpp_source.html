156  int64_t copy_term(const entry& index_entry)
157  {

167  int64_t merge_term(std::vector<entry>& indices)
168  {
169  // Write the term.
170  std::string term = indices.front().current_term();
171  terms_out_ << term << std::endl;
172 
173  // Accumulate offsets.
174  doc_ids_off_.push_back(doc_offset_);
175  doc_counts_off_.push_back(count_offset_);
176 
177  if (indices.size() == 1) { return copy_term(indices.front()); }
178 
179  // Sort by shift, i.e., effectively document IDs.
180  std::sort(indices.begin(),
181  indices.end(),
182  [](const entry& lhs, const entry& rhs) {
183  return lhs.shift() < rhs.shift();
184  });
185 
186  // Merge the posting lists.
187  int64_t occurrences = 0;
188  std::vector<document_type> doc_ids;
189  std::vector<frequency_type> doc_counts;
190  for (const entry& e : indices) {
191  occurrences += e.index()->term_occurrences(e.current_term_id());
192  auto d = e.documents();
193  auto f = e.frequencies();
194  doc_ids.insert(doc_ids.end(), d.begin(), d.end());
195  doc_counts.insert(doc_counts.end(), f.begin(), f.end());
196  }
197 
198  // Accumulate the term's document frequency.
199  term_dfs_.push_back(doc_ids.size());
200 
201  // Write documents and counts.
202  index::block_list_builder<document_type, document_codec_type, true>
203  doc_list_builder(block_size_);
204  for (const auto& doc : doc_ids) { doc_list_builder.add(doc); }
205  doc_offset_ += doc_list_builder.write(doc_ids_);
206 
207  index::block_list_builder<frequency_type, frequency_codec_type, false>
208  count_list_builder(block_size_);
209  for (const auto& count : doc_counts) { count_list_builder.add(count); }
210  count_offset_ += count_list_builder.write(doc_counts_);
211 
212  return occurrences;
213  }

216  int64_t merge_terms()
217  {
218  // Initialize heap: everyone starts with the first term.
219  std::vector<std::unique_ptr<std::ifstream>> term_streams;
220  document_type shift(0);
221  int index_num = 0;
222  BOOST_LOG_TRIVIAL(debug) << "Initializing heap..." << std::flush;
223  for (const index_type& index : indices_) {
224  term_streams.push_back(std::make_unique<std::ifstream>(
225  (sources_[index_num].dir() / "terms.txt").c_str()));
226  std::string current_term;
227  *term_streams.back() >> current_term;
228  heap_.emplace_back(index_num, 0, &index, shift, current_term);
229  shift += static_cast<document_type>(index.collection_size());
230  index_num++;
231  }
232 
233  BOOST_LOG_TRIVIAL(debug) << "Initialized.";
234  int64_t all_occurrences = 0;
235  std::vector<int64_t> occurrences;
236  term_id_type term_id = 0;
237  while (not heap_.empty())
238  {
239  std::vector<entry> indices_to_merge = indices_with_next_term();
240  BOOST_LOG_TRIVIAL(debug) << "Merging term #" << term_id++
241  << " from " << indices_to_merge.size()
242  << " indices" << std::flush;
243  occurrences.push_back(merge_term(indices_to_merge));
244  all_occurrences += occurrences.back();
245  for (entry& e : indices_to_merge) {
246  if (e.current_term_id() + 1 < e.term_count())
247  {
248  std::string current_term;
249  *term_streams[e.index_id()] >> current_term;
250  heap_.emplace_back(e.index_id(),
251  e.current_term_id() + 1,
252  e.index(),
253  e.shift(),
254  current_term);
255  std::push_heap(heap_.begin(), heap_.end());
256  }
257  }
258  }
259  // Write occurrences
260  io::dump(build_compact_table<>(occurrences),
261  index::term_occurrences_path(target_dir_));
262 
263  // Write offsets
264  io::dump(build_offset_table<>(doc_ids_off_),
265  index::doc_ids_off_path(target_dir_));
266  io::dump(build_offset_table<>(doc_counts_off_),
267  index::doc_counts_off_path(target_dir_));
268 
269  // Write term dfs.
270  auto compact_term_dfs =
271  irk::build_compact_table<frequency_type>(term_dfs_);
272  irk::io::dump(compact_term_dfs, index::term_doc_freq_path(target_dir_));
273 
274  return all_occurrences;
275  }

287  std::pair<int32_t, double> merge_sizes()
288  {
289  std::vector<int32_t> sizes;
290  std::ofstream sout(index::doc_sizes_path(target_dir_).c_str());
291  int64_t sizes_sum = 0;
292  for (const auto& index : indices_) {
293  for (int32_t doc = 0; doc < index.collection_size(); ++doc)
294  {
295  int32_t size = index.document_size(
296  static_cast<document_type>(doc));
297  sizes.push_back(size);
298  sizes_sum += size;
299  }
300  }
301  irk::compact_table<int32_t> size_table = irk::build_compact_table(
302  sizes, false);
303  sout << size_table;
304  return std::make_pair(static_cast<int32_t>(sizes.size()),
305  static_cast<double>(sizes_sum) / sizes.size());
306  }
