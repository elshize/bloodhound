149  long merge_term(std::vector<entry>& indices)
150  {
151  // Write the term.
152  std::string term = indices.front().current_term();
153  terms_out_ << term << std::endl;
154 
155  // Sort by shift, i.e., effectively document IDs.
156  std::sort(indices.begin(),
157  indices.end(),
158  [](const entry& lhs, const entry& rhs) {
159  return lhs.shift() < rhs.shift();
160  });
161 
162  // Merge the posting lists.
163  long occurrences = 0;
164  std::vector<document_type> doc_ids;
165  std::vector<frequency_type> doc_counts;
166  for (const entry& e : indices) {
167  occurrences += e.index()->term_occurrences(e.current_term_id());
168  auto pr = e.postings();
169  for (const auto& p : pr) {
170  doc_ids.push_back(p.document() + e.shift());
171  doc_counts.push_back(p.payload());
172  }
173  }
174 
175  // Accumulate the term's document frequency.
176  term_dfs_.push_back(doc_ids.size());
177 
178  // Accumulate offsets.
179  doc_ids_off_.push_back(doc_offset_);
180  doc_counts_off_.push_back(count_offset_);
181 
182  // Write documents and counts.
183  index::block_list_builder<document_type, true> doc_list_builder(
184  block_size_, document_codec_);
185  for (const auto& doc : doc_ids) { doc_list_builder.add(doc); }
186  doc_offset_ += doc_list_builder.write(doc_ids_);
187 
188  index::block_list_builder<frequency_type, false> count_list_builder(
189  block_size_, frequency_codec_);
190  for (const auto& count : doc_counts) { count_list_builder.add(count); }
191  count_offset_ += count_list_builder.write(doc_counts_);
192 
193  return occurrences;
194  }

197  long merge_terms()
198  {
199  // Initialize heap: everyone starts with the first term.
200  std::vector<std::ifstream*> term_streams;
201  document_type shift(0);
202  int index_num = 0;
203  BOOST_LOG_TRIVIAL(debug) << "Initializing heap...";
204  for (const index_type& index : indices_) {
205  term_streams.push_back(new std::ifstream(
206  (sources_[index_num].dir() / "terms.txt").c_str()));
207  std::string current_term;
208  *term_streams.back() >> current_term;
209  heap_.emplace_back(index_num, 0, &index, shift, current_term);
210  shift += index.collection_size();
211  index_num++;
212  }
213 
214  BOOST_LOG_TRIVIAL(debug) << "Initialized.";
215  long all_occurrences = 0;
216  std::vector<long> occurrences;
217  long term_id = 0;
218  while (!heap_.empty()) {
219  std::vector<entry> indices_to_merge = indices_with_next_term();
220  BOOST_LOG_TRIVIAL(debug) << "Merging term #" << term_id++
221  << " from " << indices_to_merge.size()
222  << " indices";
223  occurrences.push_back(merge_term(indices_to_merge));
224  all_occurrences += occurrences.back();
225  for (entry& e : indices_to_merge) {
226  if (std::size_t(e.current_term_id() + 1) < e.term_count())
227  {
228  std::string current_term;
229  *term_streams[e.index_id()] >> current_term;
230  heap_.emplace_back(e.index_id(),
231  e.current_term_id() + 1,
232  e.index(),
233  e.shift(),
234  current_term);
235  std::push_heap(heap_.begin(), heap_.end());
236  }
237  }
238  }
239  // Write occurrences
240  io::dump(build_compact_table<>(occurrences),
241  index::term_occurrences_path(target_dir_));
242 
243  // Write offsets
244  io::dump(build_offset_table<>(doc_ids_off_),
245  index::doc_ids_off_path(target_dir_));
246  io::dump(build_offset_table<>(doc_counts_off_),
247  index::doc_counts_off_path(target_dir_));
248 
249  // Write term dfs.
250  auto compact_term_dfs =
251  irk::build_compact_table<frequency_type>(term_dfs_);
252  irk::io::dump(compact_term_dfs, index::term_doc_freq_path(target_dir_));
253 
254  // Close other streams.
255  for (int idx = 0; idx < term_streams.size(); idx++) {
256  delete term_streams[idx];
257  }
258  terms_out_.close();
259  doc_ids_.close();
260  doc_counts_.close();
261 
262  return all_occurrences;
263  }

275  std::pair<long, double> merge_sizes()
276  {
277  std::vector<long> sizes;
278  std::ofstream sout(index::doc_sizes_path(target_dir_).c_str());
279  long sizes_sum = 0;
280  for (const auto& index : indices_) {
281  for (long doc = 0; doc < index.collection_size(); ++doc)
282  {
283  long size = index.document_size(doc);
284  sizes.push_back(size);
285  sizes_sum += size;
286  }
287  }
288  irk::compact_table<long> size_table = irk::build_compact_table(sizes, false);
289  sout << size_table;
290  return std::make_pair(sizes.size(), (double)sizes_sum / sizes.size());
291  }
