340  std::streamsize offset,

343  irk::vbyte_codec<offset_t> vb;

377 template<class Scorer, class DataSourceT>
378 void score_index(fs::path dir_path,
379  unsigned int bits,
380  std::optional<double> max = std::nullopt)
381 {
382  std::string name(typename Scorer::tag_type{});
383  fs::path scores_path = dir_path / (name + ".scores");
384  fs::path score_offsets_path = dir_path / (name + ".offsets");
385  DataSourceT source(dir_path);
386  inverted_index_view index(&source);
387 
388  int64_t offset = 0;
389  std::vector<std::size_t> offsets;
390  std::ofstream sout(scores_path.c_str());
391  std::ofstream offout(score_offsets_path.c_str());
392 
393  double max_score;
394  if (max.has_value()) {
395  max_score = max.value();
396  BOOST_LOG_TRIVIAL(info)
397  << "Max score provided: " << max_score << std::flush;
398  }
399  else {
400  BOOST_LOG_TRIVIAL(info) << "Calculating max score." << std::flush;
401  max_score = 0;
402  for (term_id_t term_id = 0; term_id < index.terms().size(); term_id++)
403  {
404  auto scorer = index.term_scorer<Scorer>(term_id);
405  for (const auto& posting : index.postings(term_id))
406  {
407  double score = scorer(
408  posting.payload(), index.document_size(posting.document()));
409  max_score = std::max(max_score, score);
410  ASSERT(score >= 0.0);
411  }
412  }
413  BOOST_LOG_TRIVIAL(info) << "Max score: " << max_score << std::flush;
414  }
415 
416  BOOST_LOG_TRIVIAL(info) << "Scoring..." << std::flush;
417  int64_t max_int = (1u << bits) - 1u;
418  for (term_id_t term_id = 0; term_id < index.terms().size(); term_id++)
419  {
420  offsets.push_back(offset);
421  irk::index::block_list_builder<std::uint32_t,
422  stream_vbyte_codec<std::uint32_t>,
423  false>
424  list_builder(index.skip_block_size());
425  auto scorer = index.term_scorer<Scorer>(term_id);
426  for (const auto& posting : index.postings(term_id))
427  {
428  double score = scorer(
429  posting.payload(), index.document_size(posting.document()));
430  auto quantized_score = static_cast<int64_t>(
431  (static_cast<double>(max_int) / max_score) * score);
432  ASSERT(quantized_score >= 0);
433  ASSERT(quantized_score <= max_int);
434  list_builder.add(quantized_score);
435  }
436  offset += list_builder.write(sout);
437  }
438  irk::offset_table<> offset_table = irk::build_offset_table<>(offsets);
439  offout << offset_table;
440 };
