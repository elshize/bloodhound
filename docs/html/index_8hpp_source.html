385  template<class Scorer>
386  void score_index(fs::path dir_path, int bits)
387  {
388  std::string name(typename Scorer::tag_type{});
389  fs::path scores_path = dir_path / (name + ".scores");
390  fs::path score_offsets_path = dir_path / (name + ".offsets");
391  v2::inverted_index_mapped_data_source source(dir_path);
392  irk::v2::inverted_index_view index(&source,
393  irk::varbyte_codec<long>{},
394  irk::varbyte_codec<long>{},
395  irk::varbyte_codec<long>{});
396 
397  long collection_size = index.collection_size();
398  long offset = 0;
399  std::vector<std::size_t> offsets;
400  std::ofstream sout(scores_path.c_str());
401  std::ofstream offout(score_offsets_path.c_str());
402 
403  double max_score = 0;
404  for (long term_id = 0; term_id < index.terms().size(); term_id++)
405  {
406  Scorer scorer = index.term_scorer<Scorer>(term_id);
407  for (const auto& posting : index.postings(term_id))
408  {
409  double score = scorer(
410  posting.payload(), index.document_size(posting.document()));
411  max_score = std::max(max_score, score);
412  }
413  }
414 
415  long max_int = (1 << bits) - 1;
416  for (long term_id = 0; term_id < index.terms().size(); term_id++)
417  {
418  offsets.push_back(offset);
419  irk::index::block_list_builder<long, false> list_builder(
420  index.skip_block_size(), irk::varbyte_codec<long>{});
421  Scorer scorer = index.term_scorer<Scorer>(term_id);
422  for (const auto& posting : index.postings(term_id))
423  {
424  double score = scorer(
425  posting.payload(), index.document_size(posting.document()));
426  long quantized_score = (long)(score * max_int / max_score);
427  list_builder.add(quantized_score);
428  }
429  offset += list_builder.write(sout);
430  }
431  irk::offset_table<> offset_table = irk::build_offset_table<>(offsets);
432  offout << offset_table;
433  }

523  template<class T>
524  std::vector<T> decode_range(term_id_type term_id,
525  const std::vector<char>& data_container,
526  const mapped_offset_table<>& offset_table,
527  bool delta) const
528  {
529  if (term_id >= offset_table.size()) {
530  throw std::out_of_range(
531  "Requested term ID out of range; requested: "
532  + std::to_string(term_id) + " but must be [0, "
533  + std::to_string(term_map_.size()) + ")");
534  }
535  auto[offset, range_size] =
536  locate(term_id, offset_table, data_container.size());
537  return delta
538  ? irk::decode_delta<varbyte_codec<T>>(gsl::span<const char>(
539  data_container.data() + offset, range_size))
540  : irk::decode<varbyte_codec<T>>(gsl::span<const char>(
541  data_container.data() + offset, range_size));
542  }
