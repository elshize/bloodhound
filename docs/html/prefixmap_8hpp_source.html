69 class prefix_map {

89  void encode_unary(std::uint32_t n)
90  {
91  for (std::uint32_t idx = 0; idx < n; ++idx) {
92  bitp_.set(1);
93  ++bitp_;
94  }
95  bitp_.set(0);
96  ++bitp_;
97  pos_ += n + 1;
98  }
99 
100  bool can_encode(std::uint32_t length) const
101  {
102  return pos_ + length <= size_ * 8;
103  }
104 
105  public:
106  block_builder(Index first_index,
107  char* block_begin,
108  std::size_t size,
109  const std::shared_ptr<hutucker_codec<char>> codec)
110  : first_index_(first_index),
111  count_(0),
112  pos_(block_data_offset * 8),
113  last_(""),
114  size_(size),
115  block_begin_(block_begin),
116  bitp_(block_begin + block_data_offset),
117  codec_(codec)
118  {
119  std::memcpy(block_begin, &first_index, sizeof(Index));
120  }
121 
122  bool add(const std::string& value)
123  {
124  assert(value.size() > 0);
125  std::uint32_t pos = 0;
126  for (; pos < last_.size(); ++pos) {
127  if (last_[pos] != value[pos]) {
128  break;
129  }
130  }
131  auto encoded = codec_->encode(value.begin() + pos, value.end());
132  if (!can_encode(value.size() + 2 + encoded.size())) {
133  return false;
134  }
135  encode_unary(pos);
136  encode_unary(value.size() - pos);
137  irk::bitcpy(bitp_, encoded);
138  bitp_ += encoded.size();
139  pos_ += encoded.size();
140  count_++;
141  last_ = value;
142  return true;
143  }
144 
145  void expand_by(std::size_t nbytes) { size_ += nbytes; }
146 
147  void reset(char* new_begin)
148  {
149  block_begin_ = new_begin;
150  bitp_ = bitptr(block_begin_ + block_data_offset);
151  }
152 
153  auto size() const { return size_; }
154 
155  void write_count()
156  {
157  std::memcpy(block_begin_ + sizeof(Index), &count_, sizeof(Counter));
158  }
159 
160  void close() { pos_ = size_ * 8; }
161  };

193  std::size_t read_unary()
194  {
195  std::size_t val = 0;
196  while (*current) {
197  ++val;
198  ++current;
199  }
200  ++current;
201  return val;
202  }
203 
204  std::string next()
205  {
206  std::size_t common_prefix_len = read_unary();
207  std::size_t suffix_len = read_unary();
208  last_value.resize(common_prefix_len);
209  std::ostringstream o(std::move(last_value), std::ios_base::ate);
210  auto reader = current.reader();
211  codec_->decode(reader, o, suffix_len);
212  last_value = o.str();
213  return last_value;
214  }

317  std::ostream& dump_leaders(std::ostream& out) const
318  {
319  raxIterator iter;
320  raxStart(&iter, block_leaders_->c_rax());
321  int result = raxSeek(&iter, "^", NULL, 0);
322  if (result == 0) {
323  throw std::bad_alloc();
324  }
325  std::string last;
326  std::vector<Index> values;
327  std::vector<std::string> keys;
328  while(raxNext(&iter)) {
329  std::string key(iter.key, iter.key + iter.key_len);
330  values.push_back(*reinterpret_cast<Index*>(iter.data));
331  keys.push_back(std::move(key));
332  }
333  auto num_values = values.size();
334  out.write(reinterpret_cast<char*>(&num_values), sizeof(num_values));
335  out.write(reinterpret_cast<char*>(values.data()),
336  values.size() * sizeof(Index));
337  for (const std::string& key : keys) {
338  out << key << '\n';
339  }
340  raxStop(&iter);
341  return out;
342  }

369  // TODO: get rid of duplication -- leaving for testing now
370  prefix_map(fs::path file,
371  const std::shared_ptr<irk::hutucker_codec<char>> codec,
372  std::size_t block_size = 1024)
373  : size_(0),
374  block_size_(block_size),
375  block_count_(0),
376  codec_(codec),
377  block_leaders_(new radix_tree<Index>())
378  {
379  std::ifstream in(file.c_str());
380  Index index(0);
381  std::string item;
382  if (!std::getline(in, item)) {
383  throw std::invalid_argument("prefix map cannot be empty");
384  }
385  block_leaders_->insert(item, index);
386  auto current_block = append_block(index, nullptr);
387  if (!current_block->add(item)) {
388  throw std::runtime_error("TODO: first item too long; feature pending");
389  }
390  index++;
391 
392  while (std::getline(in, item)) {
393  if (!current_block->add(item)) {
394  block_leaders_->insert(item, block_count_);
395  current_block = append_block(index, current_block.get());
396  if (!current_block->add(item)) {
397  while (!current_block->add(item)) {
398  expand_block(current_block.get());
399  }
400  current_block->close();
401  }
402  }
403  ++index;
404  }
405  size_ = index;
406  current_block->write_count();
407  in.close();
408  }

410  template<class StringRange>
411  prefix_map(const StringRange& items,
412  const std::shared_ptr<irk::hutucker_codec<char>> codec,
413  std::size_t block_size = 1024)
414  : size_(0),
415  block_size_(block_size),
416  block_count_(0),
417  codec_(codec),
418  block_leaders_(new radix_tree<Index>())
419  {
420  auto it = items.cbegin();
421  auto last = items.cend();
422  if (it == last) {
423  throw std::invalid_argument("prefix map cannot be empty");
424  }
425 
426  Index index(0);
427  std::string item(it->begin(), it->end());
428  block_leaders_->insert(item, 0);
429  auto current_block = append_block(index, nullptr);
430 
431  for (; it != last; ++it) {
432  std::string item(it->begin(), it->end());
433  if (!current_block->add(item)) {
434  block_leaders_->insert(item, block_count_);
435  current_block = append_block(index, current_block.get());
436  if (!current_block->add(item)) {
437  while (!current_block->add(item)) {
438  expand_block(current_block.get());
439  }
440  current_block->close();
441  }
442  }
443  ++index;
444  }
445  size_ = index;
446  current_block->write_count();
447  }

449  std::optional<Index> operator[](const std::string& key) const
450  {
451  auto block_opt = block_leaders_->seek_le(key);
452  if (!block_opt.has_value()) {
453  return std::nullopt;
454  }
455  std::size_t block_number = block_opt.value();
456  block_ptr block{blocks_.data() + block_number * block_size_, codec_};
457  Index idx = block.first_index();
458  std::string v = block.next();
459  std::uint32_t c = 1;
460  while (c < block.count() && v < key) {
461  v = block.next();
462  ++idx;
463  ++c;
464  }
465  return v == key ? std::make_optional(idx) : std::nullopt;
466  }

468  std::string operator[](const Index& val) const
469  {
470  assert(val < size_);
471  if (reverse_lookup_.empty()) { init_reverse_lookup(); }
472  auto block_pos = std::prev(std::upper_bound(
473  reverse_lookup_.begin(), reverse_lookup_.end(), val));
474  auto block_number = std::distance(reverse_lookup_.begin(), block_pos);
475  block_ptr block{blocks_.data() + block_number * block_size_, codec_};
476  Index idx = block.first_index();
477  std::string v = block.next();
478  std::uint32_t c = 1;
479  while (c < block.count() && idx < val) {
480  v = block.next();
481  ++idx;
482  ++c;
483  }
484  return v;
485  }
