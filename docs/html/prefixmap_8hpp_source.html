78  {

120  bool add(const std::string& value)
121  {
122  std::uint32_t pos = 0;
123  for (; pos < last_.size(); ++pos) {
124  if (last_[pos] != value[pos]) {
125  break;
126  }
127  }
128  auto encoded = codec_->encode(value.begin() + pos, value.end());
129  if (!can_encode(value.size() + 2 + encoded.size())) {
130  return false;
131  }
132  encode_unary(pos);
133  encode_unary(value.size() - pos);
134  irk::bitcpy(bitp_, encoded);
135  bitp_ += encoded.size();
136  pos_ += encoded.size();
137  count_++;
138  last_ = value;
139  return true;
140  }

142  void write_count()
143  {
144  std::memcpy(block_begin_ + sizeof(Index), &count_, sizeof(Counter));
145  }

187  {

200  MemoryBuffer blocks_;
201  std::size_t block_size_;

203  const std::shared_ptr<irk::coding::hutucker_codec<char>> codec_;

207  std::shared_ptr<block_builder>
208  append_block(Index index, block_builder* old_builder)
209  {
210  if (old_builder != nullptr) {
211  old_builder->write_count();
212  }
213  blocks_.resize(blocks_.size() + block_size_);
214  ++block_count_;
215  char* new_block_begin = blocks_.data() + blocks_.size() - block_size_;
216  return std::make_shared<block_builder>(
217  index, new_block_begin, block_size_, codec_);
218  }

220 public:
221  template<class StringRange>
222  prefix_map(StringRange items,
223  const std::shared_ptr<irk::coding::hutucker_codec<char>> codec,
224  std::size_t block_size = 1024)
225  : block_size_(block_size), block_count_(0), codec_(codec)
226  {
227  auto it = items.cbegin();
228  auto last = items.cend();
229  if (it == last) {
230  throw std::invalid_argument("prefix map cannot be empty");
231  }
232 
233  Index index(0);
234  std::string item = *it;
235  block_leaders_.insert(codec_->encode(item.begin(), item.end()), 0);
236  auto current_block = append_block(index, nullptr);
237 
238  for (; it != last; ++it) {
239  std::string item = *it;
240  if (!current_block->add(item)) {
241  block_leaders_.insert(
242  codec_->encode(item.begin(), item.end()), block_count_);
243  current_block = append_block(index, current_block.get());
244  if (!current_block->add(item)) {
245  throw std::runtime_error(
246  "strings longer than blocks are not supported");
247  }
248  }
249  ++index;
250  }
251  current_block->write_count();
252  }
253 
254  std::optional<Index> operator[](const std::string& key) const
255  {
256  auto encoded = codec_->encode(key.cbegin(), key.cend());
257  auto[exact, block_node] = block_leaders_.find_or_first_lower(encoded);
258  if (block_node == nullptr) {
259  return std::nullopt;
260  }
261  std::size_t block_number = block_node->value.value();
262  block_ptr block{blocks_.data() + block_number * block_size_, codec_};
263  Index idx = block.first_index();
264  if (exact) {
265  return std::make_optional(block.first_index());
266  }
267  std::string v = block.next();
268  std::uint32_t c = 1;
269  while (c < block.count() && v < key) {
270  v = block.next();
271  ++idx;
272  ++c;
273  }
274  return v == key ? std::make_optional(idx) : std::nullopt;
275  }

279 prefix_map<Index, MemoryBuffer>
280 build_prefix_map(StringRange items, std::size_t buffer_size = 1024)
281 {
282  std::vector<std::size_t> frequencies(256, 0);
283  for (const std::string& item : items) {
284  assert(item.size() > 0);
285  for (const char& ch : item) {
286  ++frequencies[ch];
287  }
288  }
289  auto codec =
290  std::make_shared<irk::coding::hutucker_codec<char>>(frequencies);
291  return prefix_map<Index, MemoryBuffer>(items, codec, buffer_size);
