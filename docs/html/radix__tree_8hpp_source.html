49 /* Get the node auxiliary data. */
50 inline void* raxGetData(raxNode* n)
51 {
52  if (n->isnull != 0) { return nullptr; }
53  auto** ndata = reinterpret_cast<void**>(
54  reinterpret_cast<char*>(n) + raxNodeCurrentLength(n) - sizeof(void*));
55  void *data;
56  std::memcpy(&data, ndata, sizeof(data));
57  return data;
58 }
