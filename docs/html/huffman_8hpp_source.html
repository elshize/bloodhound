137  template<class Symbol = char>
138  std::vector<std::size_t> symbol_frequencies(std::istream& stream)
139  {
140  std::size_t symbol_size = sizeof(Symbol);
141  std::size_t alphabet_size = 1 << (symbol_size * 8);
142  std::vector<std::size_t> frequencies(alphabet_size, 0);
143  Symbol symbol;
144  while (stream.read(reinterpret_cast<char*>(&symbol), symbol_size)) {
145  ++frequencies[static_cast<std::size_t>(symbol)];
146  }
147  return frequencies;
148  }

151  template<class Symbol = char>
152  std::list<std::shared_ptr<node<Symbol>>>
153  init_nodes(const std::vector<std::size_t>& frequencies)
154  {
155  std::vector<std::shared_ptr<node<Symbol>>> terminals;
156  for (std::size_t symbol_number = 0; symbol_number < frequencies.size();
157  ++symbol_number) {
158  std::size_t frequency = frequencies[symbol_number];
159  if (frequency > 0) {
160  Symbol symbol = static_cast<Symbol>(symbol_number);
161  terminals.push_back(make_terminal(symbol, frequency));
162  }
163  }
164  std::sort(terminals.begin(),
165  terminals.end(),
166  [](const auto& lhs, const auto& rhs) {
167  return lhs->symbol < rhs->symbol;
168  });
169  std::list<std::shared_ptr<node<Symbol>>> terminal_list(
170  terminals.begin(), terminals.end());
171  return terminal_list;
172  }
