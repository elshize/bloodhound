57  template<class InputIterator, class OutputIterator>
58  std::ptrdiff_t encode(InputIterator in, OutputIterator out) const
59  {
60  using iterator_type = std::iterator_traits<InputIterator>;
61  using value_type =
62  std::make_unsigned_t<typename iterator_type::value_type>;
63  auto v = static_cast<value_type>(*in);
64  std::ptrdiff_t size = 0;
65  while (true)
66  {
67  auto last_byte = v < maxv;
68  *out = v % maxv + maxv * last_byte;
69  ++out;
70  ++size;
71  if (last_byte) { break; }
72  v /= maxv;
73  }
74  return size;
75  }

103  InputIterator decode(InputIterator in, OutputIterator out) const
104  {

108  unsigned char b;

113  std::size_t val = b & 0b01111111u;
114  n |= val << shift;
