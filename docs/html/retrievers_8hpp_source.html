688  ranges::view::zip(lists_for_terms, term_weights)) {

694

697  stats_["terms"] = lists_for_terms.size();
698 
699  std::vector<PostingList> postlists = lists_for_terms;
700  auto postings = run_for("len", compare_len, postlists, term_weights, k, threshold);
701  run_for("ms", compare_maxscore, postlists, term_weights, k, threshold);
702  stats_["postings"] = postings;
703 
704  return results;
705  }

769  std::vector<std::size_t> sorted_by_length(
770  const std::vector<PostingList>& lists_for_terms)
771  {
772  std::vector<std::size_t> sorted;
773  for (std::size_t idx = 0; idx < lists_for_terms.size(); ++idx) {
774  if (lists_for_terms[idx].length() > 0) {
775  sorted.push_back(idx);
776  }
777  }
778  std::sort(sorted.begin(),
779  sorted.end(),
780  [lists_for_terms](const std::size_t& a, const std::size_t& b) {
781  return lists_for_terms[a].length()
782  < lists_for_terms[b].length();
783  });
784  return sorted;
785  }
786 
787  virtual std::vector<Result> retrieve(
788  const std::vector<PostingList>& lists_for_terms,
789  const std::vector<Score>& term_weights,
790  std::size_t k)
791  {
792  std::vector<Score> acc(collection_size, Score(0));
793  irkit::Heap<Score, Doc, std::less<Score>, std::unordered_map<Doc, int>>
794  top_results(k);
795 
796  auto postlists = sorted_by_length(lists_for_terms);
797 
798  Score threshold = Score(0);
799  Score remaining_max = Score(0);
800  for (auto& term : postlists) {
801  remaining_max += lists_for_terms[term].max_score;
802  }
803 
804  std::size_t postings = 0;
805  std::size_t idx = 0;
806  while (idx < postlists.size() && threshold <= remaining_max) {
807  auto& postlist = lists_for_terms[idx];
808  auto dociter = postlist.doc_begin();
809  auto scoreiter = postlist.score_begin();
810  for (; dociter != postlist.doc_end(); ++dociter, ++scoreiter) {
811  auto doc = *dociter;
812  auto score_delta = *scoreiter * term_weights[idx];
813  acc[doc] += score_delta;
814  top_results.push_with_limit(acc[doc], doc, k);
815  }
816  remaining_max -= postlist.max_score;
817  if (top_results.size() == k) {
818  threshold = top_results.top().key;
819  }
820  postings += postlist.length();
821  idx++;
822  }
823 
824  std::list<Doc> visited;
825  for (Doc doc = Doc(0); doc < Doc(acc.size()); ++doc) {
826  if (acc[doc] > Score(0)) {
827  visited.push_back(doc);
828  }
829  }
830 
831  std::size_t first_phase = postings;
832  std::size_t second_phase = 0;
833  std::size_t next_ge_count = 0;
834  std::size_t removed = 0;
835  for (; idx < postlists.size(); ++idx) {
836  auto& postlist = lists_for_terms[idx];
837  auto iter = postlist.begin();
838  auto end = postlist.end();
839  auto visited_iter = visited.begin();
840  auto visited_end = visited.end();
841  while (visited_iter != visited_end) {
842  Doc doc = *visited_iter;
843  ++next_ge_count;
844  iter = postlist.next_ge(iter, doc);
845  if (iter == end) {
846  break;
847  }
848  if (iter->doc == doc) {
849  acc[doc] += iter->score;
850  if (acc[doc] + remaining_max < threshold) {
851  visited_iter = visited.erase(visited_iter);
852  ++removed;
853  continue;
854  } else {
855  top_results.push_with_limit(acc[doc], doc, k);
856  if (top_results.size() == k) {
857  threshold = top_results.top().key;
858  }
859  }
860  }
861  ++visited_iter;
862  }
863  remaining_max -= postlist.max_score;
864  second_phase += postlist.length();
865  }
866  std::cout << first_phase
867  << "," << second_phase
868  << "," << next_ge_count
869  << "," << first_phase + second_phase
870  << "," << removed
871  << std::endl;
872  return heap_to_results(top_results);
873  }
